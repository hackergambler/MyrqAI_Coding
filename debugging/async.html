Debugging Async AI API Calls

Fix common async/await errors in Python when calling AI APIs: event loop bugs, race conditions, and timeout handling.

Common Async Bugs with AI APIs

Async code multiplies the ways things can go wrong. These are the errors developers encounter most often when building async AI applications.

Bug 1: Running Async in Sync Context

❌ RuntimeError: This event loop is already running
Python - broken code
# WRONG: calling asyncio.run() inside Jupyter or FastAPI
import asyncio
from openai import AsyncOpenAI

async def get_answer(q):
    client = AsyncOpenAI()
    resp = await client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": q}]
    )
    return resp.choices[0].message.content

# Crashes in Jupyter because the loop is already running
result = asyncio.run(get_answer("hello"))
✅ Use nest_asyncio in Jupyter, await directly in FastAPI
Python - fixed code
# In Jupyter: install and apply nest_asyncio
import nest_asyncio
nest_asyncio.apply()
result = asyncio.run(get_answer("hello"))

# In FastAPI: just await directly, no asyncio.run()
from fastapi import FastAPI
app = FastAPI()

@app.get("/")
async def endpoint(q: str):
    return {"answer": await get_answer(q)}  # no asyncio.run()!

Bug 2: Forgetting await

❌ TypeError: object coroutine cannot be used in await
Python - broken code
# WRONG: missing await returns a coroutine object, not a result
async def summarize(text: str):
    client = AsyncOpenAI()
    resp = client.chat.completions.create(  # Missing await!
        model="gpt-4o",
        messages=[{"role": "user", "content": text}]
    )
    return resp.choices[0].message.content  # AttributeError!
✅ Always await async client methods
Python - fixed code
async def summarize(text: str):
    client = AsyncOpenAI()
    resp = await client.chat.completions.create(  # await added
        model="gpt-4o",
        messages=[{"role": "user", "content": text}]
    )
    return resp.choices[0].message.content

Bug 3: Sequential Calls Instead of Parallel

❌ Sequential despite async - wastes time
Python - slow sequential code
# WRONG: each call waits for the previous one
async def process_batch(texts: list) -> list:
    results = []
    for text in texts:
        result = await summarize(text)  # sequential!
        results.append(result)
    return results
✅ Use asyncio.gather() for true parallelism
Python - parallel with gather
import asyncio

async def process_batch(texts: list) -> list:
    tasks = [summarize(text) for text in texts]
    results = await asyncio.gather(*tasks, return_exceptions=True)
    return [
        r if not isinstance(r, Exception) else f"Error: {r}"
        for r in results
    ]
💡 Rate limiting with async

Use asyncio.Semaphore to limit concurrent API calls: sem = asyncio.Semaphore(10) then async with sem: await api_call()

Bug 4: Race Condition in Shared State

❌ Non-deterministic results under concurrent load
Python - broken shared state
# WRONG: shared mutable messages across concurrent coroutines
messages = [{"role": "system", "content": "You are helpful."}]

async def handle_request(user_msg: str) -> str:
    messages.append({"role": "user", "content": user_msg})  # RACE!
    resp = await async_client.chat.completions.create(
        model="gpt-4o-mini", messages=messages
    )
    answer = resp.choices[0].message.content
    messages.append({"role": "assistant", "content": answer})  # RACE!
    return answer
✅ Isolate state per request
Python - isolated state
async def handle_request(user_msg: str) -> str:
    # Each call gets its OWN messages list - no shared state
    local_messages = [
        {"role": "system", "content": "You are helpful."},
        {"role": "user", "content": user_msg}
    ]
    resp = await async_client.chat.completions.create(
        model="gpt-4o-mini", messages=local_messages
    )
    return resp.choices[0].message.content

Bug 5: httpx AsyncClient Not Closed — Resource Leak

❌ Broken — new client created per call, never closed
Python
# WRONG: creates a new connection pool on every call — memory leak
async def fetch_data(url: str) -> str:
    import httpx
    client = httpx.AsyncClient()           # new pool every time!
    resp = await client.get(url)
    return resp.text                       # client never closed!
✅ Fixed — use context manager or module-level client
Python
import httpx

# Option 1: context manager (closes automatically)
async def fetch_data(url: str) -> str:
    async with httpx.AsyncClient() as client:
        resp = await client.get(url, timeout=10.0)
        return resp.text

# Option 2: module-level client reused across calls (more efficient)
_client: httpx.AsyncClient | None = None

async def get_client() -> httpx.AsyncClient:
    global _client
    if _client is None:
        _client = httpx.AsyncClient(timeout=10.0, limits=httpx.Limits(max_connections=100))
    return _client

Bug 6: Semaphore Deadlock With Nested Async Calls

❌ Broken — deadlock when semaphore scope wraps another semaphore call
Python
import asyncio
sem = asyncio.Semaphore(2)

async def inner_call():
    async with sem:  # tries to acquire same semaphore!
        return await llm.complete("inner")

async def outer_call():
    async with sem:  # holds the semaphore...
        return await inner_call()  # ...then tries to acquire it again: DEADLOCK!
✅ Fixed — separate semaphores per level, or don't nest
Python
# Use separate semaphores for different call levels
outer_sem = asyncio.Semaphore(5)   # for outer orchestration calls
inner_sem = asyncio.Semaphore(10)  # for inner tool calls

async def inner_call():
    async with inner_sem:
        return await llm.complete("inner")

async def outer_call():
    async with outer_sem:  # different semaphore — no deadlock
        return await inner_call()

Bug 7: asyncio.gather() Silently Swallows Exceptions

❌ Broken — one task failure is hidden from the caller
Python
# WRONG: if one task raises, gather re-raises but other results are LOST
results = await asyncio.gather(task1(), task2(), task3())
# If task2 raises ValueError, you get the error but task1/task3 results vanish
✅ Fixed — use return_exceptions=True to handle partial failures
Python
results = await asyncio.gather(
    task1(), task2(), task3(),
    return_exceptions=True  # exceptions become values, not raised
)

# Handle each result individually
for i, result in enumerate(results):
    if isinstance(result, Exception):
        print(f"Task {i} failed: {result}")
    else:
        print(f"Task {i} result: {result}")

Bug 8: Background Task Exception Silently Swallowed

❌ Broken — asyncio.create_task() exceptions are silently lost
Python
async def broken():
    task = asyncio.create_task(some_job())  # fire and forget
    await asyncio.sleep(5)
    # If some_job() raises, exception is silently discarded!
✅ Fixed — add done callback to surface exceptions
Python
import asyncio, logging

def log_task_errors(task: asyncio.Task) -> None:
    try:
        task.result()  # re-raises stored exception
    except asyncio.CancelledError:
        pass
    except Exception as e:
        logging.error(f"Background task failed: {e}", exc_info=True)

task = asyncio.create_task(some_job())
task.add_done_callback(log_task_errors)  # always log failures
