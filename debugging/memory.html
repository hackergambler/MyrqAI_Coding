Debugging Memory Issues in AI Apps

Fix GPU out-of-memory errors, context window overflows, Python RAM exhaustion, and LangChain conversation memory leaks in production AI systems.

Why AI Apps Run Out of Memory

Memory bugs are especially common in AI development because models are large, context windows grow with every turn, and embeddings accumulate silently. Here are the most frequent causes and their fixes.

Bug 1: GPU Out-of-Memory (CUDA OOM)

❌ RuntimeError: CUDA out of memory. Tried to allocate 2.50 GiB
Python - broken code
# WRONG: loads full fp32 weights, no batching limit, keeps all gradients
from transformers import AutoModelForCausalLM, AutoTokenizer
import torch

model = AutoModelForCausalLM.from_pretrained("meta-llama/Llama-3-8b")
model = model.to("cuda")

def generate(prompts: list) -> list:
    inputs = tokenizer(prompts, return_tensors="pt").to("cuda")
    outputs = model.generate(**inputs, max_new_tokens=512)
    return tokenizer.batch_decode(outputs)
✓ Load in 4-bit, disable gradients, process in small batches
Python - fixed code
from transformers import AutoModelForCausalLM, AutoTokenizer, BitsAndBytesConfig
import torch

# Load in 4-bit quantization - cuts VRAM use by ~75%
bnb_config = BitsAndBytesConfig(load_in_4bit=True, bnb_4bit_compute_dtype=torch.float16)
model = AutoModelForCausalLM.from_pretrained(
    "meta-llama/Llama-3-8b",
    quantization_config=bnb_config,
    device_map="auto"
)

def generate(prompts: list, batch_size: int = 4) -> list:
    results = []
    for i in range(0, len(prompts), batch_size):
        batch = prompts[i:i + batch_size]
        inputs = tokenizer(batch, return_tensors="pt", padding=True).to("cuda")
        with torch.no_grad():
            outputs = model.generate(**inputs, max_new_tokens=256)
        results.extend(tokenizer.batch_decode(outputs, skip_special_tokens=True))
        torch.cuda.empty_cache()
    return results

Bug 2: Context Window Overflow

❌ openai.BadRequestError: This model's maximum context length is 128000 tokens
Python - broken code
# WRONG: appends every message forever - context grows without limit
conversation_history = []

def chat(user_message: str) -> str:
    conversation_history.append({"role": "user", "content": user_message})
    response = client.chat.completions.create(
        model="gpt-4o",
        messages=conversation_history  # after 100+ turns this explodes
    )
    reply = response.choices[0].message.content
    conversation_history.append({"role": "assistant", "content": reply})
    return reply
✓ Sliding window - keep system prompt and last N messages
Python - fixed code
import tiktoken

SYSTEM_PROMPT = {"role": "system", "content": "You are a helpful assistant."}
MAX_TOKENS = 100_000
conversation_history = []

def count_tokens(messages: list) -> int:
    enc = tiktoken.encoding_for_model("gpt-4o")
    return sum(len(enc.encode(m["content"])) for m in messages)

def trim_history(history: list) -> list:
    while count_tokens([SYSTEM_PROMPT] + history) > MAX_TOKENS and len(history) > 2:
        history = history[2:]  # drop oldest user+assistant pair
    return history

def chat(user_message: str) -> str:
    conversation_history.append({"role": "user", "content": user_message})
    trimmed = trim_history(conversation_history.copy())
    response = client.chat.completions.create(
        model="gpt-4o",
        messages=[SYSTEM_PROMPT] + trimmed
    )
    reply = response.choices[0].message.content
    conversation_history.append({"role": "assistant", "content": reply})
    return reply
💡 Token Counting Rule of Thumb

1 token is roughly 4 characters in English. Always track token usage with tiktoken before hitting the API limit. Leave a buffer of at least 20% for the model's output.

Bug 3: LangChain ConversationBufferMemory Leak

❌ Shared memory accumulates all users' messages - OOM after many sessions
Python - broken code
from langchain.memory import ConversationBufferMemory
from langchain.chains import ConversationChain

# WRONG: one shared memory object for ALL users
shared_memory = ConversationBufferMemory()
chain = ConversationChain(llm=llm, memory=shared_memory)

def handle_request(user_id: str, message: str) -> str:
    return chain.predict(input=message)  # all users share same history!
✓ Per-session memory with token limit via ConversationTokenBufferMemory
Python - fixed code
from langchain.memory import ConversationTokenBufferMemory
from langchain.chains import ConversationChain

sessions: dict = {}

def get_chain(user_id: str) -> ConversationChain:
    if user_id not in sessions:
        memory = ConversationTokenBufferMemory(
            llm=llm,
            max_token_limit=4000  # auto-trims old messages
        )
        sessions[user_id] = ConversationChain(llm=llm, memory=memory)
    return sessions[user_id]

def handle_request(user_id: str, message: str) -> str:
    return get_chain(user_id).predict(input=message)

def evict_session(user_id: str):
    sessions.pop(user_id, None)  # call this on session end to free RAM

Bug 4: Embedding Vectors Filling RAM

❌ MemoryError: Unable to allocate 12 GiB for array
Python - broken code
import numpy as np

# WRONG: 500k docs x 1536 floats x 4 bytes = ~3 GB stored in a Python list
def embed_all(docs: list) -> np.ndarray:
    embeddings = []
    for doc in docs:
        resp = client.embeddings.create(model="text-embedding-3-small", input=doc)
        embeddings.append(resp.data[0].embedding)
    return np.array(embeddings, dtype=np.float32)  # crashes here
✓ Stream embeddings to disk with numpy memmap or use a vector DB
Python - fixed code
import numpy as np

EMBEDDING_DIM = 1536
BATCH_SIZE = 100

def embed_to_disk(docs: list, path: str = "embeddings.npy") -> np.memmap:
    # Memory-mapped file: data lives on disk, not in RAM
    arr = np.memmap(path, dtype=np.float32, mode="w+",
                    shape=(len(docs), EMBEDDING_DIM))
    for i in range(0, len(docs), BATCH_SIZE):
        batch = docs[i:i + BATCH_SIZE]
        resp = client.embeddings.create(
            model="text-embedding-3-small", input=batch)
        for j, item in enumerate(resp.data):
            arr[i + j] = item.embedding
    arr.flush()
    return arr

# Or use a vector DB (Chroma, Pinecone, Qdrant) which handles
# persistence and efficient retrieval automatically.

Bug 5: Hugging Face Model Not Released from VRAM

❌ Loading a second model fails - first model still holds GPU memory
Python - broken code
# WRONG: model goes out of scope but CUDA memory is not freed
def run_experiment():
    model = AutoModelForCausalLM.from_pretrained("mistralai/Mistral-7B-v0.1").cuda()
    outputs = model.generate(...)
    # model deleted here but CUDA allocator still holds the pages

run_experiment()
# Fails: CUDA OOM because VRAM was never returned
model2 = AutoModelForCausalLM.from_pretrained("meta-llama/Llama-3-8b").cuda()
✓ Delete the model, collect garbage, then empty the cache
Python - fixed code
import torch, gc

def run_experiment():
    model = AutoModelForCausalLM.from_pretrained("mistralai/Mistral-7B-v0.1").cuda()
    try:
        outputs = model.generate(...)
    finally:
        del model                 # remove Python reference
        gc.collect()              # run Python garbage collector
        torch.cuda.empty_cache()  # release cached CUDA blocks

run_experiment()
model2 = AutoModelForCausalLM.from_pretrained("meta-llama/Llama-3-8b").cuda()  # works!
⚠ torch.cuda.empty_cache() Does Not Free Everything

empty_cache() releases the PyTorch allocator's cache back to CUDA, but memory held by live tensors is unaffected. Always del the model first, run gc.collect(), then call empty_cache().

Bug 6: Streaming Chunks Accumulating in a Global List

❌ RAM grows continuously during long-running streaming sessions
Python - broken code
# WRONG: global list keeps every chunk from every call forever
all_chunks = []

def stream_response(prompt: str) -> str:
    stream = client.chat.completions.create(
        model="gpt-4o",
        messages=[{"role": "user", "content": prompt}],
        stream=True
    )
    for chunk in stream:
        text = chunk.choices[0].delta.content or ""
        all_chunks.append(text)  # accumulates forever in production!
    return "".join(all_chunks)
✓ Use a local variable or generator - never accumulate across calls
Python - fixed code
# Option A: collect locally, garbage collected after each call
def stream_response(prompt: str) -> str:
    chunks = []  # local variable, not global
    stream = client.chat.completions.create(
        model="gpt-4o",
        messages=[{"role": "user", "content": prompt}],
        stream=True
    )
    for chunk in stream:
        chunks.append(chunk.choices[0].delta.content or "")
    return "".join(chunks)

# Option B: generator - zero accumulation, ideal for SSE / WebSocket
def stream_tokens(prompt: str):
    stream = client.chat.completions.create(
        model="gpt-4o",
        messages=[{"role": "user", "content": prompt}],
        stream=True
    )
    for chunk in stream:
        yield chunk.choices[0].delta.content or ""
💡 Quick Diagnostics

Use torch.cuda.memory_summary() for GPU stats, tracemalloc for Python heap leaks, and psutil.Process().memory_info().rss to monitor RAM over time in production.
