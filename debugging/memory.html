solution_context_management.py 
import tiktoken

def count_tokens(messages, model="gpt-4"):
    """Count tokens in message array"""
    encoding = tiktoken.encoding_for_model(model)
    num_tokens = 0
    for message in messages:
        num_tokens += 4  # message overhead
        for key, value in message.items():
            num_tokens += len(encoding.encode(value))
    return num_tokens + 2  # reply priming

def trim_conversation(messages, max_tokens=100000, model="gpt-4"):
    """Trim conversation to fit within token limit"""
    # Always keep system message
    system_msg = [m for m in messages if m["role"] == "system"]
    other_msgs = [m for m in messages if m["role"] != "system"]
    
    # Keep most recent messages that fit
    trimmed = []
    current_tokens = count_tokens(system_msg, model)
    
    for msg in reversed(other_msgs):
        msg_tokens = count_tokens([msg], model)
        if current_tokens + msg_tokens < max_tokens:
            trimmed.insert(0, msg)
            current_tokens += msg_tokens
        else:
            break
    
    return system_msg + trimmed

# Usage
messages = trim_conversation(messages, max_tokens=100000)
response = client.chat.completions.create(
    model="gpt-4",
    messages=messages
)
💡 Pro Tip: Implement a sliding window with summarization for long conversations. Periodically summarize older messages and replace them with the summary to preserve context while reducing tokens.

def calculate_safe_context(
    model_max_tokens: int,
    desired_output_tokens: int,
    safety_margin: float = 0.1
) -> int:
    """Calculate safe input token limit"""
    # Reserve space for output and safety margin
    reserved = desired_output_tokens + int(model_max_tokens * safety_margin)
    return model_max_tokens - reserved

# Model context limits
MODEL_LIMITS = {
    "gpt-4-turbo": 128000,
    "gpt-4o": 128000,
    "claude-3-opus": 200000,
    "claude-3-sonnet": 200000,
    "claude-3-haiku": 200000,
    "gemini-1.5-pro": 1000000,
}

def prepare_messages_for_model(messages, model, max_output=4096):
    """Prepare messages with safe token limits"""
    model_limit = MODEL_LIMITS.get(model, 100000)
    safe_input = calculate_safe_context(model_limit, max_output)
    
    return trim_conversation(messages, max_tokens=safe_input)

Error Message

'system', 'user', 'assistant', 'tool', but got 'AI'.

Root Causes

  • Incorrect role names (using "AI" instead of "assistant")
  • Missing required fields in message objects
  • Messages not alternating properly (user/assistant/user)
  • Tool messages without corresponding tool_call_id

solution_message_validation.py 
from typing import List, Dict, Any
from pydantic import BaseModel, field_validator
from enum import Enum

class MessageRole(str, Enum):
    SYSTEM = "system"
    USER = "user"
    ASSISTANT = "assistant"
    TOOL = "tool"

class Message(BaseModel):
    role: MessageRole
    content: str
    name: str | None = None
    tool_call_id: str | None = None
    tool_calls: List[Dict] | None = None
    
    @field_validator('content')
    @classmethod
    def content_not_empty(cls, v):
        if not v or not v.strip():
            raise ValueError('Content cannot be empty')
        return v

def validate_messages(messages: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    """Validate and normalize message array"""
    validated = []
    
    # Role normalization map
    role_map = {
        "ai": "assistant",
        "bot": "assistant",
        "human": "user",
        "person": "user",
    }
    
    for i, msg in enumerate(messages):
        # Normalize role
        role = msg.get("role", "").lower()
        role = role_map.get(role, role)
        
        # Validate
        try:
            validated_msg = Message(
                role=role,
                content=msg.get("content", ""),
                name=msg.get("name"),
                tool_call_id=msg.get("tool_call_id"),
                tool_calls=msg.get("tool_calls")
            )
            validated.append(validated_msg.model_dump(exclude_none=True))
        except Exception as e:
            print(f"Invalid message at index {i}: {e}")
            continue
    
    return validated

# Usage
messages = validate_messages(raw_messages)
response = client.chat.completions.create(
    model="gpt-4",
    messages=messages

solution_embedding_consistency.py 
import chromadb
from openai import OpenAI

# Embedding model dimensions
EMBEDDING_DIMS = {
    "text-embedding-ada-002": 1536,
    "text-embedding-3-small": 1536,
    "text-embedding-3-large": 3072,
}

class ConsistentEmbedder:
    def __init__(self, model="text-embedding-3-small"):
        self.client = OpenAI()
        self.model = model
        self.dimension = EMBEDDING_DIMS[model]
    
    def embed(self, texts: list[str]) -> list[list[float]]:
        response = self.client.embeddings.create(
            model=self.model,
            input=texts
        )
        return [e.embedding for e in response.data]
    
    def create_collection(self, chroma_client, name: str):
        """Create collection with correct dimensions"""
        return chroma_client.create_collection(
            name=name,
            metadata={
                "embedding_model": self.model,
                "dimension": self.dimension
            }
        )
    
    def validate_collection(self, collection) -> bool:
        """Check if collection matches embedder"""
        meta = collection.metadata or {}
        return meta.get("dimension") == self.dimension

# Usage - always use same embedder instance
embedder = ConsistentEmbedder(model="text-embedding-3-small")
chroma = chromadb.Client()

# Create new collection
collection = embedder.create_collection(chroma, "my_docs")

# Or validate existing
if not embedder.validate_collection(existing_collection):
    raise ValueError("Collection dimension mismatch - recreate with current model")

Chunking Destroys Semantic Meaning

Problem 
# Poor retrieval results because chunks break mid-sentence
# or split related information across chunks

Root Causes

  • Fixed-size chunking ignoring sentence/paragraph boundaries
  • No overlap between chunks losing context
  • Chunk size too small for semantic meaning
  • Not using semantic or recursive chunking

Solutions

solution_semantic_chunking.py 
from langchain.text_splitter import (
    RecursiveCharacterTextSplitter,
    MarkdownHeaderTextSplitter,
    TokenTextSplitter
)
import tiktoken

class SmartChunker:
    def __init__(self, chunk_size=1000, chunk_overlap=200):
        self.chunk_size = chunk_size
        self.chunk_overlap = chunk_overlap
        
        # Recursive splitter respects natural boundaries
        self.text_splitter = RecursiveCharacterTextSplitter(
            chunk_size=chunk_size,
            chunk_overlap=chunk_overlap,
            separators=[
                "\n\n\n",   # Major sections
                "\n\n",      # Paragraphs
                "\n",        # Lines
                ". ",        # Sentences
                "! ",
                "? ",
                "; ",
                ", ",        # Clauses
                " ",         # Words
                ""           # Characters (last resort)
            ],
            length_function=self._count_tokens
        )
        
        # Markdown-aware splitter
        self.md_splitter = MarkdownHeaderTextSplitter(
            headers_to_split_on=[
                ("#", "h1"),
                ("##", "h2"),
                ("###", "h3"),
            ]
        )
    
    def _count_tokens(self, text: str) -> int:
        enc = tiktoken.get_encoding("cl100k_base")
        return len(enc.encode(text))
    
    def chunk_text(self, text: str) -> list[str]:
        """Chunk plain text with semantic awareness"""
        return self.text_splitter.split_text(text)
    
    def chunk_markdown(self, markdown: str) -> list[dict]:
        """Chunk markdown preserving headers as metadata"""
        # First split by headers
        header_chunks = self.md_splitter.split_text(markdown)
        
        # Then recursively split large sections
        final_chunks = []
        for chunk in header_chunks:
            if self._count_tokens(chunk.page_content) > self.chunk_size:
                sub_chunks = self.text_splitter.split_text(chunk.page_content)
                for sub in sub_chunks:
                    final_chunks.append({
                        "content": sub,
                        "metadata": chunk.metadata
                    })
            else:
                final_chunks.append({
                    "content": chunk.page_content,
                    "metadata": chunk.metadata
                })
        
        return final_chunks
    
    def chunk_code(self, code: str, language: str = "python") -> list[str]:
        """Chunk code preserving function/class boundaries"""
        from langchain.text_splitter import Language, RecursiveCharacterTextSplitter
        
        lang_map = {
            "python": Language.PYTHON,
            "javascript": Language.JS,
            "typescript": Language.TS,
            "java": Language.JAVA,
        }
        
        splitter = RecursiveCharacterTextSplitter.from_language(
            language=lang_map.get(language, Language.PYTHON),
            chunk_size=self.chunk_size,
            chunk_overlap=self.chunk_overlap
        )
        
        return splitter.split_text(code)

# Usage
chunker = SmartChunker(chunk_size=1000, chunk_overlap=200)

# For documents
chunks = chunker.chunk_text(document_text)

# For markdown docs
md_chunks = chunker.chunk_markdown(readme_content)

# For code
code_chunks = chunker.chunk_code(python_code, "python")

solution_persistent_memory.py 
import json
import redis
from datetime import datetime, timedelta
from typing import Optional
import hashlib

class PersistentMemory:
    def __init__(self, redis_url: str = "redis://localhost:6379"):
        self.redis = redis.from_url(redis_url)
        self.ttl = timedelta(days=30)  # Memory retention period
    
    def _session_key(self, user_id: str, session_id: str) -> str:
        return f"memory:{user_id}:{session_id}"
    
    def save_conversation(
        self, 
        user_id: str, 
        session_id: str, 
        messages: list[dict]
    ) -> None:
        """Save conversation to Redis"""
        key = self._session_key(user_id, session_id)
        data = {
            "messages": messages,
            "updated_at": datetime.utcnow().isoformat(),
            "message_count": len(messages)
        }
        self.redis.setex(
            key, 
            self.ttl, 
            json.dumps(data)
        )
    
    def load_conversation(
        self, 
        user_id: str, 
        session_id: str
    ) -> Optional[list[dict]]:
        """Load conversation from Redis"""
        key = self._session_key(user_id, session_id)
        data = self.redis.get(key)
        if data:
            return json.loads(data)["messages"]
        return None
    
    def save_user_preferences(self, user_id: str, prefs: dict) -> None:
        """Save user preferences"""
        key = f"prefs:{user_id}"
        self.redis.set(key, json.dumps(prefs))
    
    def get_user_preferences(self, user_id: str) -> dict:
        """Get user preferences"""
        key = f"prefs:{user_id}"
        data = self.redis.get(key)
        return json.loads(data) if data else {}
    
    def get_recent_sessions(self, user_id: str, limit: int = 10) -> list[str]:
        """Get user's recent session IDs"""
        pattern = f"memory:{user_id}:*"
        keys = self.redis.keys(pattern)
        
        # Sort by update time
        sessions = []
        for key in keys:
            data = self.redis.get(key)
            if data:
                parsed = json.loads(data)
                sessions.append({
                    "session_id": key.decode().split(":")[-1],
                    "updated_at": parsed["updated_at"]
                })
        
        sessions.sort(key=lambda x: x["updated_at"], reverse=True)
        return [s["session_id"] for s in sessions[:limit]]

from langchain.memory import ConversationBufferMemory
from langchain.schema import HumanMessage, AIMessage

class PersistentLangChainMemory(ConversationBufferMemory):
    def __init__(self, user_id: str, session_id: str, **kwargs):
        super().__init__(**kwargs)
        self.storage = PersistentMemory()
        self.user_id = user_id
        self.session_id = session_id

    def _load(self):
        """Load from persistent storage"""
        messages = self.storage.load_conversation(
            self.user_id, 
            self.session_id
        )
        if messages:
            for msg in messages:
                if msg["role"] == "user":
                    self.chat_memory.add_user_message(msg["content"])
                else:
                    self.chat_memory.add_ai_message(msg["content"])
    
    def save_context(self, inputs, outputs):
        """Save and persist"""
        super().save_context(inputs, outputs)
        
        # Persist to Redis
        messages = [
            {"role": "user" if isinstance(m, HumanMessage) else "assistant", 
             "content": m.content}
            for m in self.chat_memory.messages
        ]
        self.storage.save_conversation(
            self.user_id, 
            self.session_id, 
            messages
        )

Retrieval Returning Irrelevant Context

Problem 
# Vector search returns documents that don't answer the question
# Top-k results are semantically similar but not relevant
# Context pollution degrading response quality

Root Causes

  • Embedding similarity ≠ relevance for the query
  • k value too high bringing in noise
  • No re-ranking or filtering of results
  • Query and documents in different semantic spaces

Solutions

solution_hybrid_retrieval.py 
from sentence_transformers import CrossEncoder
from typing import List, Tuple
import numpy as np

class HybridRetriever:
    def __init__(self, vector_store, reranker_model="cross-encoder/ms-marco-MiniLM-L-6-v2"):
        self.vector_store = vector_store
        self.reranker = CrossEncoder(reranker_model)
    
    def retrieve(
        self, 
        query: str, 
        k: int = 5, 
        initial_k: int = 20,
        score_threshold: float = 0.5
    ) -> List[dict]:
        """Hybrid retrieval with re-ranking"""
        
        # Step 1: Initial broad retrieval
        candidates = self.vector_store.similarity_search_with_score(
            query, 
            k=initial_k
        )
        
        # Step 2: Re-rank with cross-encoder
        if candidates:
            pairs = [(query, doc.page_content) for doc, _ in candidates]
            rerank_scores = self.reranker.predict(pairs)
            
            # Combine with original scores
            results = []
            for (doc, vec_score), rerank_score in zip(candidates, rerank_scores):
                if rerank_score > score_threshold:
                    results.append({
                        "document": doc,
                        "vector_score": float(vec_score),
                        "rerank_score": float(rerank_score),
                        "combined_score": 0.3 * vec_score + 0.7 * rerank_score
                    })
            
            # Sort by combined score and return top k
            results.sort(key=lambda x: x["combined_score"], reverse=True)
            return results[:k]
        
        return []
    
    def retrieve_with_mmr(
        self, 
        query: str, 
        k: int = 5, 
        diversity: float = 0.5
    ) -> List[dict]:
        """Maximal Marginal Relevance for diverse results"""
        return self.vector_store.max_marginal_relevance_search(
            query,
            k=k,
            fetch_k=k * 4,
            lambda_mult=diversity  # 0=max diversity, 1=max relevance
        )

# Usage with filtering
retriever = HybridRetriever(vector_store)

# Get relevant, high-quality context
results = retriever.retrieve(
    query="How do I handle rate limiting in the OpenAI API?",
    k=5,
    score_threshold=0.6  # Filter low-relevance results
)

# Build context from results
context = "\n\n".join([
    r["document"].page_content 
    for r in results
])

import tiktoken
from anthropic import Anthropic

class UniversalTokenCounter:
    """Count tokens for any major LLM provider"""
    
    # Model to encoding mapping
    OPENAI_ENCODINGS = {
        "gpt-4": "cl100k_base",
        "gpt-4-turbo": "cl100k_base",
        "gpt-4o": "o200k_base",
        "gpt-3.5-turbo": "cl100k_base",
        "text-embedding-3-small": "cl100k_base",
        "text-embedding-3-large": "cl100k_base",
    }
    
    def __init__(self):
        self._encoders = {}
        self._anthropic = Anthropic()
    
    def _get_encoder(self, encoding: str):
        if encoding not in self._encoders:
            self._encoders[encoding] = tiktoken.get_encoding(encoding)
        return self._encoders[encoding]
    
    def count_openai(self, text: str, model: str = "gpt-4") -> int:
        """Count tokens for OpenAI models"""
        encoding = self.OPENAI_ENCODINGS.get(model, "cl100k_base")
        encoder = self._get_encoder(encoding)
        return len(encoder.encode(text))
    
    def count_anthropic(self, text: str) -> int:
        """Count tokens for Claude models"""
        return self._anthropic.count_tokens(text)
    
    def count_messages_openai(self, messages: list, model: str = "gpt-4") -> int:
        """Count tokens in message array for OpenAI"""
        encoding = self.OPENAI_ENCODINGS.get(model, "cl100k_base")
        encoder = self._get_encoder(encoding)
        
        tokens = 0
        for msg in messages:
            tokens += 4  # Message overhead
            for key, value in msg.items():
                if isinstance(value, str):
                    tokens += len(encoder.encode(value))
                elif isinstance(value, list):  # Tool calls
                    tokens += len(encoder.encode(str(value)))
        
        tokens += 2  # Reply priming
        return tokens
    
    def estimate_cost(
        self, 
        input_tokens: int, 
        output_tokens: int, 
        model: str
    ) -> float:
        """Estimate API cost"""
        # Prices per 1M tokens (as of 2024)
        PRICES = {
            "gpt-4-turbo": {"input": 10, "output": 30},
            "gpt-4o": {"input": 5, "output": 15},
            "gpt-4o-mini": {"input": 0.15, "output": 0.6},
            "claude-3-opus": {"input": 15, "output": 75},
            "claude-3-sonnet": {"input": 3, "output": 15},
            "claude-3-haiku": {"input": 0.25, "output": 1.25},
        }
        
        prices = PRICES.get(model, {"input": 1, "output": 1})
        return (
            (input_tokens / 1_000_000) * prices["input"] +
            (output_tokens / 1_000_000) * prices["output"]
        )

# Usage
counter = UniversalTokenCounter()

# Count for different providers
openai_tokens = counter.count_openai(text, "gpt-4o")
claude_tokens = counter.count_anthropic(text)

# Estimate costs
cost = counter.estimate_cost(10000, 2000, "gpt-4o")
print(f"Estimated cost: ${cost:.4f}")

Common Memory Bugs — With Fixes

Bug 1: Context Not Passed Between Agent Steps

Each LLM call starts fresh with no history — the agent cannot remember what happened in previous steps.

Python - broken
# WRONG: every step forgets everything
def agent_step(task: str) -> str:
    resp = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": task}]  # no history!
    )
    return resp.choices[0].message.content

agent_step("The user is Alice, age 30.")
agent_step("What is the user's name?")  # has no idea — returns wrong answer

✅ Fix — accumulate history across steps

Python - fixed
history = []

def agent_step(task: str) -> str:
    history.append({"role": "user", "content": task})
    resp = client.chat.completions.create(
        model="gpt-4o-mini", messages=history
    )
    answer = resp.choices[0].message.content
    history.append({"role": "assistant", "content": answer})
    return answer

Bug 2: Vector Store Query Returns Empty — Documents Never Added

Vector store created but documents never indexed — all similarity queries return empty results.

Python - broken
# WRONG: forgot to call from_documents() or add_documents()
from langchain_community.vectorstores import Chroma
from langchain_openai import OpenAIEmbeddings

vectorstore = Chroma(embedding_function=OpenAIEmbeddings())
# documents never added!
results = vectorstore.similarity_search("Python tutorial")  # returns []

✅ Fix — use from_documents() or verify count before querying

Python - fixed
from langchain_core.documents import Document

docs = [Document(page_content="Python is great for AI"), ...]

# Build index from documents
vectorstore = Chroma.from_documents(docs, OpenAIEmbeddings())
print(f"Indexed: {vectorstore._collection.count()} docs")  # always verify

Bug 3: LangChain Memory Not Persisted — Lost on Restart

In-memory ConversationBufferMemory is lost every time the app restarts. Users lose their entire conversation history.

Python - broken
from langchain.memory import ConversationBufferMemory
memory = ConversationBufferMemory()  # in-RAM only — gone on restart
memory.save_context({"input": "hi"}, {"output": "hello!"})
# App restarts... conversation lost

✅ Fix — use Redis-backed or SQLite-backed message history

Python - fixed
import os
from langchain_community.chat_message_histories import RedisChatMessageHistory

history = RedisChatMessageHistory(
    session_id="user_alice_session_1",
    url=os.environ["REDIS_URL"],
    ttl=86400  # expire after 24 hours
)
# Survives restarts — stored in Redis
history.add_user_message("What is Python?")
history.add_ai_message("Python is a programming language.")
print(f"History has {len(history.messages)} messages")

Bug 4: Duplicate Context From Semantic Retrieval

The same document chunk is retrieved multiple times because it matches multiple aspects of the query, wasting tokens and confusing the LLM.

Python - broken
# WRONG: no deduplication — same chunk appears 3x
docs = retriever.invoke("Python decorators")
context = "\n\n".join(d.page_content for d in docs)
# context repeats "Python decorators are functions..." three times

✅ Fix — deduplicate by content hash before building context

Python - fixed
def deduplicate_docs(docs: list) -> list:
    seen, unique = set(), []
    for doc in docs:
        h = hash(doc.page_content[:200])
        if h not in seen:
            seen.add(h); unique.append(doc)
    return unique

docs = retriever.invoke("Python decorators")
context = "\n\n".join(d.page_content for d in deduplicate_docs(docs))
print(f"Using {len(deduplicate_docs(docs))} unique chunks")

Bug 5: Embedding Cache Miss — Same Text Re-Embedded Every Call

No caching on embedding calls means identical text strings are sent to the API repeatedly, wasting money.

Python - broken
# WRONG: same system prompt embedded on every single request
def get_embedding(text: str) -> list:
    resp = client.embeddings.create(model="text-embedding-3-small", input=text)
    return resp.data[0].embedding  # called 1000x with same text = $$ wasted

✅ Fix — cache embeddings with functools.lru_cache or Redis

Python - fixed
from functools import lru_cache
import hashlib, json

# In-memory LRU cache (fast, lost on restart)
@lru_cache(maxsize=1000)
def get_embedding_cached(text: str) -> tuple:
    resp = client.embeddings.create(model="text-embedding-3-small", input=text)
    return tuple(resp.data[0].embedding)  # tuple is hashable for lru_cache

# Check cache stats
print(get_embedding_cached.cache_info())
