Lab 1: Infinite Token Generation Loop

Symptom: API calls never stop, token bill skyrockets, app hangs indefinitely.

❌ Broken - infinite loop that never terminates
Python
def generate_until_done(prompt: str) -> str:
    result = ""
    while True:  # no exit condition!
        resp = client.chat.completions.create(
            model="gpt-4o-mini",
            messages=[{"role":"user","content":prompt + result}]
        )
        chunk = resp.choices[0].message.content
        result += chunk
        if resp.choices[0].finish_reason == "stop":
            break  # BUG: always True on single call, never actually loops!
    return result  # but if you remove break, it loops forever
✅ Use max_tokens as a hard limit - no loop needed
Python
def generate(prompt: str, max_tokens: int = 2048) -> str:
    resp = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role":"user","content":prompt}],
        max_tokens=max_tokens  # hard limit always
    )
    return resp.choices[0].message.content

Lab 2: Context Window Overflow

Symptom: openai.BadRequestError: This model maximum context length is 128000 tokens. Your messages resulted in 145231 tokens.

❌ Broken - chat history grows forever
Python
class ChatBot:
    def __init__(self):
        self.history = []  # grows forever, never trimmed!

    def chat(self, msg: str) -> str:
        self.history.append({"role":"user","content":msg})
        resp = client.chat.completions.create(
            model="gpt-4o-mini", messages=self.history
        )
        answer = resp.choices[0].message.content
        self.history.append({"role":"assistant","content":answer})
        return answer  # crashes after ~100 long messages
✅ Sliding window with token counting
Python
import tiktoken

class ChatBot:
    def __init__(self, max_tokens: int = 80_000):
        self.history = []
        self.max_tokens = max_tokens
        self.enc = tiktoken.encoding_for_model("gpt-4o-mini")

    def _trim(self):
        while len(self.history) > 2:
            tokens = sum(len(self.enc.encode(str(m))) for m in self.history)
            if tokens <= self.max_tokens: break
            self.history.pop(0)  # remove oldest message

    def chat(self, msg: str) -> str:
        self.history.append({"role":"user","content":msg})
        self._trim()
        resp = client.chat.completions.create(
            model="gpt-4o-mini", messages=self.history
        )
        answer = resp.choices[0].message.content
        self.history.append({"role":"assistant","content":answer})
        return answer

Lab 3: Race Condition in Async Agent

Symptom: Non-deterministic results. Tool results get applied to the wrong requests. Output is garbled under concurrent load.

❌ Broken - shared mutable state across coroutines
Python
# WRONG: all concurrent requests share the same messages list
messages = [{"role":"system","content":"You are helpful."}]

async def handle(user_msg: str) -> str:
    messages.append({"role":"user","content":user_msg})  # RACE!
    resp = await async_client.chat.completions.create(
        model="gpt-4o-mini", messages=messages
    )
    answer = resp.choices[0].message.content
    messages.append({"role":"assistant","content":answer})  # RACE!
    return answer

# Two concurrent requests will corrupt each other history
await asyncio.gather(handle("Hello"), handle("Hi"))
✅ Isolated state per request - never share mutable state
Python
async def handle(user_msg: str) -> str:
    # Each call gets its OWN private messages list
    local_messages = [
        {"role":"system","content":"You are helpful."},
        {"role":"user","content":user_msg}
    ]
    resp = await async_client.chat.completions.create(
        model="gpt-4o-mini", messages=local_messages
    )
    return resp.choices[0].message.content

Symptom: RAG returns irrelevant results, or a dimension mismatch error when loading a saved vector store.

❌ Broken - different models for indexing vs querying
Python

vs = FAISS.from_documents(docs, OpenAIEmbeddings(model="text-embedding-3-large"))
vs.save_local("./index")

# Query with different model (1536 dims) - DIMENSION MISMATCH!

# Save
vs = FAISS.from_documents(docs, embeddings)
vs.save_local("./index")
Path("./index/meta.json").write_text(json.dumps({"model": EMBED_MODEL}))

# Load with assertion
meta = json.loads(Path("./index/meta.json").read_text())
assert meta["model"] == EMBED_MODEL, f"Model mismatch: {meta['model']}"

Lab 5: Tool Result Not JSON-Serializable

Symptom: TypeError: Object of type datetime is not JSON serializable when passing tool results back to the model.

❌ Broken - returning non-serializable types from tools
Python
from datetime import datetime
from decimal import Decimal

def get_order(order_id: str) -> dict:
    return {
        "id": order_id,
        "created_at": datetime.now(),   # NOT serializable!
        "amount": Decimal("49.99"),      # NOT serializable!
    }

result = get_order("ORD-123")
json.dumps(result)  # TypeError crash!
✅ Return only JSON-safe types from tools
Python
def get_order(order_id: str) -> dict:
    return {
        "id": order_id,
        "created_at": datetime.now().isoformat(),  # str
        "amount": float(Decimal("49.99")),           # float
    }

# Universal safe serializer for any tool result
def safe_json(obj) -> str:
    def default(o):
        if isinstance(o, datetime): return o.isoformat()
        if isinstance(o, Decimal): return float(o)
        return str(o)
    return json.dumps(obj, default=default)

Lab 6: Missing await in Streaming

Symptom: TypeError: 'coroutine' object is not an async iterable

❌ Broken - missing await before iterating stream
Python
async def stream(prompt: str):
    stream = async_client.chat.completions.create(  # missing await!
        model="gpt-4o",
        messages=[{"role":"user","content":prompt}],
        stream=True
    )
    async for chunk in stream:  # TypeError: coroutine is not async iterable
        yield chunk.choices[0].delta.content
✅ Use the .stream() context manager which handles await internally
Python
async def stream(prompt: str):
    async with async_client.chat.completions.stream(
        model="gpt-4o",
        messages=[{"role":"user","content":prompt}]
    ) as s:
        async for event in s:
            if event.choices and event.choices[0].delta.content:
                yield event.choices[0].delta.content

Lab 7: Agent Hallucinates Function Arguments

Symptom: KeyError or TypeError when executing a tool - the LLM invented argument names that do not exist in the schema.

❌ Broken - trusting LLM arguments blindly
Python
def execute_tool(tool_call):
    args = json.loads(tool_call.function.arguments)
    fn = TOOLS[tool_call.function.name]
    return fn(**args)  # KeyError if LLM invented wrong arg name
✅ Validate arguments against the schema before executing
Python
from jsonschema import validate, ValidationError

def execute_tool(tool_call) -> dict:
    name = tool_call.function.name
    try:
        args = json.loads(tool_call.function.arguments)
    except json.JSONDecodeError as e:
        return {"error": f"Invalid JSON args: {e}"}
    try:
        validate(args, TOOL_SCHEMAS[name])
    except ValidationError as e:
        return {"error": f"Invalid args for {name}: {e.message}"}
    fn = TOOLS.get(name)
    if not fn: return {"error": f"Unknown tool: {name}"}
    try:
        return fn(**args)
    except Exception as e:
        return {"error": str(e)}

Lab 8: Memory Leak in Chat Session Store

Symptom: Server RAM grows unbounded over hours, eventually getting OOM-killed by the OS.

❌ Broken - global dict that never cleans up old sessions
Python
# WRONG: grows forever as users come and go
sessions: dict[str, list] = {}  # never evicted!

def get_session(session_id: str) -> list:
    if session_id not in sessions:
        sessions[session_id] = []
    return sessions[session_id]
✅ TTL cache with automatic expiry
Python
from cachetools import TTLCache
from threading import Lock

# Max 1000 sessions, each expires after 30 minutes of inactivity
sessions = TTLCache(maxsize=1000, ttl=1800)
lock = Lock()

def get_session(session_id: str) -> list:
    with lock:
        if session_id not in sessions:
            sessions[session_id] = []
        return sessions[session_id]

# Monitor in production
def session_stats():
    return {"active": len(sessions), "max": sessions.maxsize, "ttl": sessions.ttl}

Lab 9: Rate Limit Cascade Failure

Symptom: One slow period triggers retries. The retry burst makes the rate limit worse. A death spiral consumes all retries and fails permanently.

❌ Broken - fixed retry delay causes thundering herd
Python
import time

def call_api(prompt):
    for attempt in range(5):
        try:
            resp = client.chat.completions.create(...)
            return resp.choices[0].message.content
        except RateLimitError:
            time.sleep(1)  # all threads wake at the same moment = thundering herd
    raise Exception("Failed after 5 attempts")
✅ Exponential backoff with random jitter plus circuit breaker
Python
import time, random
from openai import RateLimitError

def call_with_backoff(fn, max_retries=5):
    for attempt in range(max_retries):
        try:
            return fn()
        except RateLimitError:
            base = 2 ** attempt
            jitter = random.uniform(0, 1)  # prevents synchronized retries
            delay = min(base + jitter, 60)    # cap at 60s
            print(f"Rate limited. Waiting {delay:.1f}s (attempt {attempt+1})")
            time.sleep(delay)
    raise Exception("Max retries exceeded")

# Usage
result = call_with_backoff(lambda: client.chat.completions.create(
    model="gpt-4o-mini", messages=[{"role":"user","content":"hello"}]
))

Lab 10: JSON Parsing on Streaming Output

Symptom: json.JSONDecodeError: Expecting value when parsing LLM output as JSON during streaming - because you are parsing incomplete chunks.

❌ Broken - parsing each chunk as complete JSON
Python
import json

# WRONG: streaming chunks are fragments like "{", "\"name\":", "Alice"
with client.chat.completions.stream(
    model="gpt-4o",
    messages=[{"role":"user","content":"Return JSON: {name: Alice, age: 30}"}]
) as stream:
    for event in stream:
        chunk = event.choices[0].delta.content
        if chunk:
            data = json.loads(chunk)  # JSONDecodeError every time!
✅ Collect the full stream, then parse - display live meanwhile
Python
import json

full_text = ""
with client.chat.completions.stream(
    model="gpt-4o",
    messages=[{"role":"user","content":"Return JSON: {name: Alice, age: 30}"}],
    response_format={"type":"json_object"}  # enforce JSON output
) as stream:
    for event in stream:
        chunk = event.choices[0].delta.content or ""
        full_text += chunk
        print(chunk, end="", flush=True)  # show live

# Parse AFTER stream completes
data = json.loads(full_text)
print(f"\nParsed: name={data['name']}, age={data['age']}")
