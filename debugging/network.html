solution_rate_limiter.py 
import asyncio
import time
from typing import TypeVar, Callable
from functools import wraps
import random

T = TypeVar('T')

class RateLimiter:
    """Token bucket rate limiter with sliding window"""
    
    def __init__(
        self, 
        requests_per_minute: int = 60,
        tokens_per_minute: int = 90000
    ):
        self.rpm_limit = requests_per_minute
        self.tpm_limit = tokens_per_minute
        self.request_times = []
        self.token_usage = []
        self.window = 60  # seconds
    
    def _clean_old_entries(self):
        """Remove entries outside the sliding window"""
        cutoff = time.time() - self.window
        self.request_times = [t for t in self.request_times if t > cutoff]
        self.token_usage = [(t, tok) for t, tok in self.token_usage if t > cutoff]
    
    def _current_rpm(self) -> int:
        self._clean_old_entries()
        return len(self.request_times)
    
    def _current_tpm(self) -> int:
        self._clean_old_entries()
        return sum(tok for _, tok in self.token_usage)
    
    async def acquire(self, estimated_tokens: int = 1000):
        """Wait until rate limit allows request"""
        while True:
            self._clean_old_entries()
            
            # Check both limits
            rpm_ok = self._current_rpm() < self.rpm_limit
            tpm_ok = self._current_tpm() + estimated_tokens < self.tpm_limit
            
            if rpm_ok and tpm_ok:
                self.request_times.append(time.time())
                return
            
            # Calculate wait time
            if not rpm_ok:
                wait_time = self.request_times[0] + self.window - time.time()
            else:
                wait_time = self.token_usage[0][0] + self.window - time.time()
            
            await asyncio.sleep(max(0.1, wait_time + 0.1))
    
    def record_usage(self, tokens: int):
        """Record actual token usage after request"""
        self.token_usage.append((time.time(), tokens))

# Decorator with exponential backoff
def with_retry(
    max_retries: int = 5,
    base_delay: float = 1.0,
    max_delay: float = 60.0
):
    def decorator(func: Callable[..., T]) -> Callable[..., T]:
        @wraps(func)
        async def wrapper(*args, **kwargs) -> T:
            last_exception = None
            
            for attempt in range(max_retries):
                try:
                    return await func(*args, **kwargs)
                except Exception as e:
                    last_exception = e
                    error_str = str(e).lower()
                    
                    # Check if retryable
                    if any(x in error_str for x in ['rate_limit', '429', 'overloaded']):
                        # Exponential backoff with jitter
                        delay = min(base_delay * (2 ** attempt), max_delay)
                        jitter = random.uniform(0, delay * 0.1)
                        wait_time = delay + jitter
                        
                        print(f"Rate limited. Retrying in {wait_time:.2f}s (attempt {attempt + 1}/{max_retries})")
                        await asyncio.sleep(wait_time)
                    else:
                        raise  # Non-retryable error
            
            raise last_exception
        return wrapper
    return decorator

# Usage
rate_limiter = RateLimiter(requests_per_minute=50, tokens_per_minute=80000)

@with_retry(max_retries=5)
async def call_openai(messages: list, estimated_tokens: int = 1000):
    await rate_limiter.acquire(estimated_tokens)
    
    response = await client.chat.completions.create(
        model="gpt-4",
        messages=messages
    )
    
    # Record actual usage
    rate_limiter.record_usage(response.usage.total_tokens)
    return response

# Batch processing with rate limiting
async def process_batch(items: list, batch_size: int = 10):
    results = []
    
    for i in range(0, len(items), batch_size):
        batch = items[i:i + batch_size]
        tasks = [call_openai(item) for item in batch]
        batch_results = await asyncio.gather(*tasks, return_exceptions=True)
        results.extend(batch_results)
        
        # Small delay between batches
        await asyncio.sleep(0.5)
    
    return results

import asyncio
from typing import List, Optional
from dataclasses import dataclass
from enum import Enum
import time

class CircuitState(Enum):
    CLOSED = "closed"      # Normal operation
    OPEN = "open"          # Failing, reject requests
    HALF_OPEN = "half_open"  # Testing if recovered

@dataclass
class CircuitBreaker:
    """Circuit breaker pattern for API resilience"""
    
    failure_threshold: int = 5
    recovery_timeout: float = 30.0
    half_open_max_calls: int = 3
    
    def __post_init__(self):
        self.state = CircuitState.CLOSED
        self.failures = 0
        self.last_failure_time = 0
        self.half_open_calls = 0
    
    def can_execute(self) -> bool:
        if self.state == CircuitState.CLOSED:
            return True
        
        if self.state == CircuitState.OPEN:
            # Check if recovery timeout passed
            if time.time() - self.last_failure_time > self.recovery_timeout:
                self.state = CircuitState.HALF_OPEN
                self.half_open_calls = 0
                return True
            return False
        
        # HALF_OPEN - allow limited calls
        return self.half_open_calls < self.half_open_max_calls
    
    def record_success(self):
        if self.state == CircuitState.HALF_OPEN:
            self.half_open_calls += 1
            if self.half_open_calls >= self.half_open_max_calls:
                self.state = CircuitState.CLOSED
                self.failures = 0
        else:
            self.failures = 0
    
    def record_failure(self):
        self.failures += 1
        self.last_failure_time = time.time()
        
        if self.failures >= self.failure_threshold:
            self.state = CircuitState.OPEN

# Multi-provider fallback
class ResilientLLMClient:
    """Client with fallback to multiple providers"""
    
    def __init__(self):
        self.providers = {
            "openai": {
                "client": OpenAI(),
                "model": "gpt-4-turbo",
                "circuit": CircuitBreaker()
            },
            "anthropic": {
                "client": anthropic.Anthropic(),
                "model": "claude-3-sonnet-20240229",
                "circuit": CircuitBreaker()
            },
            "azure": {
                "client": AzureOpenAI(
                    azure_endpoint=os.getenv("AZURE_ENDPOINT"),
                    api_key=os.getenv("AZURE_API_KEY"),
                    api_version="2024-02-15-preview"
                ),
                "model": "gpt-4",
                "circuit": CircuitBreaker()
            }
        }
        self.primary = "openai"
    
    async def complete(
        self, 
        messages: List[dict],
        preferred_provider: Optional[str] = None
    ) -> str:
        """Complete with automatic fallback"""
        
        # Order providers by preference
        provider_order = [preferred_provider or self.primary]
        provider_order.extend([
            p for p in self.providers.keys() 
            if p not in provider_order
        ])
        
        last_error = None
        for provider_name in provider_order:
            provider = self.providers[provider_name]
            
            if not provider["circuit"].can_execute():
                print(f"{provider_name} circuit is open, skipping")
                continue
            
            try:
                result = await self._call_provider(provider_name, messages)
                provider["circuit"].record_success()
                return result
            except Exception as e:
                print(f"{provider_name} failed: {e}")
                provider["circuit"].record_failure()
                last_error = e
                continue
        
        raise Exception(f"All providers failed. Last error: {last_error}")
    
    async def _call_provider(self, name: str, messages: List[dict]) -> str:
        provider = self.providers[name]
        
        if name == "anthropic":
            # Convert messages format for Anthropic
            response = provider["client"].messages.create(
                model=provider["model"],
                max_tokens=4096,
                messages=messages
            )
            return response.content[0].text
        else:
            response = provider["client"].chat.completions.create(
                model=provider["model"],
                messages=messages
            )
            return response.choices[0].message.content

# Custom DNS resolver with fallback
class CustomDNSResolver:
    def __init__(self, nameservers: list = None):
        # Use public DNS as fallback
        self.nameservers = nameservers or [
            "8.8.8.8",        # Google
            "8.8.4.4",        # Google
            "1.1.1.1",        # Cloudflare
            "208.67.222.222", # OpenDNS
        ]
        self.cache = {}
        self.cache_ttl = 300  # 5 minutes
    
    async def resolve(self, hostname: str) -> str:
        """Resolve hostname with caching and fallback"""
        # Check cache
        if hostname in self.cache:
            ip, timestamp = self.cache[hostname]
            if time.time() - timestamp < self.cache_ttl:
                return ip
        
        # Try each nameserver
        for ns in self.nameservers:
            try:
                resolver = aiodns.DNSResolver(nameservers=[ns])
                result = await resolver.query(hostname, 'A')
                ip = result[0].host
                
                # Cache result
                self.cache[hostname] = (ip, time.time())
                return ip
            except Exception as e:
                print(f"DNS query to {ns} failed: {e}")
                continue
        
        raise Exception(f"All DNS servers failed for {hostname}")

import asyncio
from dataclasses import dataclass
from typing import Any, Callable
import uuid

@dataclass
class QueuedRequest:
    id: str
    func: Callable
    args: tuple
    kwargs: dict
    priority: int = 0
    future: asyncio.Future = None

class RequestQueue:
    """Priority queue for API requests with concurrency control"""
    
    def __init__(self, max_concurrent: int = 10):
        self.queue = asyncio.PriorityQueue()
        self.max_concurrent = max_concurrent
        self.semaphore = asyncio.Semaphore(max_concurrent)
        self.active = 0
        self._workers_started = False
    
    async def enqueue(
        self, 
        func: Callable, 
        *args, 
        priority: int = 0,
        **kwargs
    ) -> Any:
        """Add request to queue and wait for result"""
        request = QueuedRequest(
            id=str(uuid.uuid4()),
            func=func,
            args=args,
            kwargs=kwargs,
            priority=priority,
            future=asyncio.Future()
        )
        
        # Priority queue uses (priority, id) for ordering
        await self.queue.put((priority, request.id, request))
        
        # Start workers if not already running
        if not self._workers_started:
            asyncio.create_task(self._process_queue())
            self._workers_started = True
        
        return await request.future
    
    async def _process_queue(self):
        """Process queued requests"""
        while True:
            _, _, request = await self.queue.get()
            
            async with self.semaphore:
                try:
                    result = await request.func(*request.args, **request.kwargs)
                    request.future.set_result(result)
                except Exception as e:
                    request.future.set_exception(e)
                finally:
                    self.queue.task_done()

import asyncio
from dataclasses import dataclass, field
from datetime import datetime
from typing import Dict, List
import statistics

@dataclass
class HealthMetrics:
    success_count: int = 0
    failure_count: int = 0
    latencies: List[float] = field(default_factory=list)
    last_check: datetime = None
    is_healthy: bool = True
    
    @property
    def success_rate(self) -> float:
        total = self.success_count + self.failure_count
        return self.success_count / total if total > 0 else 1.0
    
    @property
    def avg_latency(self) -> float:
        return statistics.mean(self.latencies) if self.latencies else 0
    
    @property
    def p99_latency(self) -> float:
        if len(self.latencies) < 100:
            return max(self.latencies) if self.latencies else 0
        return statistics.quantiles(self.latencies, n=100)[98]

class APIHealthMonitor:
    def __init__(self):
        self.metrics: Dict[str, HealthMetrics] = {}
        self.check_interval = 60  # seconds
    
    async def check_health(self, endpoint: str, check_func) -> bool:
        """Run health check for an endpoint"""
        start = time.time()
        
        if endpoint not in self.metrics:
            self.metrics[endpoint] = HealthMetrics()
        
        try:
            await check_func()
            latency = time.time() - start
            
            self.metrics[endpoint].success_count += 1
            self.metrics[endpoint].latencies.append(latency)
            self.metrics[endpoint].is_healthy = True
            
            # Keep last 1000 latencies
            if len(self.metrics[endpoint].latencies) > 1000:
                self.metrics[endpoint].latencies = self.metrics[endpoint].latencies[-1000:]
            
            return True
        except Exception as e:
            self.metrics[endpoint].failure_count += 1
            self.metrics[endpoint].is_healthy = False
            return False
        finally:
            self.metrics[endpoint].last_check = datetime.now()
    
    def get_status(self) -> dict:
        """Get current health status"""
        return {
            endpoint: {
                "healthy": m.is_healthy,
                "success_rate": f"{m.success_rate:.2%}",
                "avg_latency_ms": f"{m.avg_latency * 1000:.0f}",
                "p99_latency_ms": f"{m.p99_latency * 1000:.0f}",
                "last_check": m.last_check.isoformat() if m.last_check else None
            }
            for endpoint, m in self.metrics.items()
        }
    
    async def start_monitoring(self, endpoints: Dict[str, callable]):
        """Start continuous health monitoring"""
        while True:
            for endpoint, check_func in endpoints.items():
                await self.check_health(endpoint, check_func)
            await asyncio.sleep(self.check_interval)

# Usage
monitor = APIHealthMonitor()

async def check_openai():
    await client.models.list()

async def check_anthropic():
    await claude_client.messages.create(
        model="claude-3-haiku-20240307",
        max_tokens=10,
        messages=[{"role": "user", "content": "hi"}]
    )

# Start monitoring
asyncio.create_task(monitor.start_monitoring({
    "openai": check_openai,
    "anthropic": check_anthropic
}))

Common Network Bugs — With Fixes

Bug 1: SSL Certificate Verification Error

SSL verification fails when calling internal APIs. The wrong fix (verify=False) is a serious security risk.

Python - broken
# Error: SSLCertVerificationError: certificate verify failed
# WRONG: disabling verification is a critical security risk
import httpx
resp = httpx.get("https://internal-api.corp.com", verify=False)  # NEVER do this!

✅ Fix — provide the correct CA bundle or fix the certificate

Python - fixed
# Option 1: provide custom CA bundle
resp = httpx.get("https://internal-api.corp.com", verify="/etc/ssl/company-ca.crt")

# Option 2: set env var for all requests
# export REQUESTS_CA_BUNDLE=/etc/ssl/company-ca.crt

# Option 3: permanent fix — add to system trust store
# sudo cp company-ca.crt /usr/local/share/ca-certificates/
# sudo update-ca-certificates

Bug 2: Connection Pool Exhaustion Under Load

Unlimited concurrent requests exhaust the HTTP connection pool, causing timeouts under normal production load.

Python - broken
# Error: httpcore.PoolTimeout: timed out waiting for connection
# WRONG: 500 concurrent tasks exhaust the default pool of 10
async def process_all(items: list):
    tasks = [process(item) for item in items]  # 500 concurrent requests!
    return await asyncio.gather(*tasks)

✅ Fix — configure pool limits and throttle with a semaphore

Python - fixed
import httpx, asyncio

limits = httpx.Limits(max_connections=50, max_keepalive_connections=20)
sem = asyncio.Semaphore(20)  # max 20 concurrent API calls

async def process_all(items: list):
    async with httpx.AsyncClient(limits=limits) as client:
        async def bounded(item):
            async with sem:
                return await process(item, client=client)
        return await asyncio.gather(*[bounded(i) for i in items])

Bug 3: Timeout Too Short for Long LLM Calls

Default HTTP timeout (5 seconds) kills in-progress LLM generation. GPT-4o can take 30–120 seconds for long outputs.

Python - broken
# Error: httpx.ReadTimeout: timed out reading response
# Default timeout is 5 seconds — GPT-4o can take 2 minutes
import httpx
resp = httpx.post("https://api.openai.com/v1/chat/completions", json={...})
# Crashes halfway through long responses

✅ Fix — set per-operation timeouts appropriate for LLM latency

Python - fixed
from openai import OpenAI

# OpenAI SDK: set generous timeout
client = OpenAI(timeout=120.0)  # 2 minutes for long completions

# httpx: fine-grained per-phase timeouts
timeout = httpx.Timeout(
    connect=5.0,    # fast fail if server unreachable
    read=120.0,     # generous read timeout for streaming
    write=10.0,
    pool=5.0
)

Bug 4: Proxy Blocks HTTPS API Calls in Corporate Environments

Works locally but fails in corporate environments where a proxy intercepts HTTPS traffic. The SDK does not automatically read proxy env vars.

Python - broken
# Works at home, breaks at the office
# HTTPS_PROXY is set in the environment but SDK ignores it
from openai import OpenAI
client = OpenAI()  # does not auto-read HTTPS_PROXY!
# Error: SSLError or ConnectionRefused

✅ Fix — pass proxy explicitly via http_client

Python - fixed
import httpx, os
from openai import OpenAI

proxy = os.environ.get("HTTPS_PROXY") or os.environ.get("https_proxy")
http_client = httpx.Client(proxy=proxy) if proxy else None
client = OpenAI(http_client=http_client)

# Test connection
try:
    client.models.list()
    print(f"Connected (proxy: {proxy or 'none'})")
except Exception as e:
    print(f"Failed: {e}")

Bug 5: Streaming Response Cuts Off Mid-Token

The stream ends abruptly because max_tokens was reached. The response is truncated without warning, causing downstream JSON parse errors.

Python - broken
# WRONG: max_tokens too low + not checking stop_reason
with client.chat.completions.stream(
    model="gpt-4o",
    messages=[{"role": "user", "content": "Generate a 500-item JSON list"}],
    max_tokens=256  # too low — cuts off the JSON mid-way!
) as stream:
    text = "".join(c for c in stream.text_stream)
data = json.loads(text)  # JSONDecodeError — incomplete JSON

✅ Fix — set adequate max_tokens and check stop_reason

Python - fixed
with client.chat.completions.stream(
    model="gpt-4o",
    messages=[{"role": "user", "content": "Generate a 500-item JSON list"}],
    max_tokens=8192  # sufficient headroom
) as stream:
    text = "".join(c for c in stream.text_stream)

final = stream.get_final_message()
if final.stop_reason == "max_tokens":
    raise ValueError(f"Response truncated! Used {final.usage.output_tokens} tokens.")
data = json.loads(text)  # safe to parse now
