import anthropic

client = anthropic.Anthropic()

message = client.messages.create(
    model="claude-sonnet-4-20250514",
    max_tokens=1024,
    messages=[
        {"role": "user", 
         "content": "Explain quantum computing"}
    ]
)

print(message.content[0].text)
