from ollama import Client

# Connect to local Ollama instance
client = Client(host='http://localhost:11434')

# Run inference on local model
response = client.chat(
    model='llama3.2:70b',
    messages=[{
        'role': 'user',
        'content': 'Explain quantum computing'
    }],
    options={
        'temperature': 0.7,
        'num_ctx': 8192,
        'num_gpu': 2  # Multi-GPU
    }
)

print(response['message']['content'])

Quick Start: Your First Local LLM

Get up and running in under 5 minutes

1

Install Ollama

bash 
# macOS / Linux
curl -fsSL https://ollama.com/install.sh | sh

# Windows - Download from ollama.com
# Or use Docker
docker run -d -v ollama:/root/.ollama -p 11434:11434 ollama/ollama
2

Pull a Model

bash 
# Pull Llama 3.2 (recommended for most use cases)
ollama pull llama3.2

# Or try smaller models for limited hardware
ollama pull phi3:mini      # 3.8B params, ~2GB
ollama pull gemma2:2b      # 2B params, ~1.5GB

# Check available models
ollama list
3

Run Inference

python 
import ollama

# Simple chat completion
response = ollama.chat(
    model='llama3.2',
    messages=[{
        'role': 'user',
        'content': 'Write a haiku about coding'
    }]
)

print(response['message']['content'])

# Streaming response
for chunk in ollama.chat(
    model='llama3.2',
    messages=[{'role': 'user', 'content': 'Tell me a story'}],
    stream=True
):
    print(chunk['message']['content'], end='', flush=True)
