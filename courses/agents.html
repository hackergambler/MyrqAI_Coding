from langchain.agents import create_react_agent
from langchain.tools import Tool
from langchain_openai import ChatOpenAI

# Define agent tools
tools = [
    Tool(
        name="web_search",
        func=search_web,
        description="Search the web for info"
    ),
    Tool(
        name="calculator",
        func=calculate,
        description="Perform math operations"
    ),
    Tool(
        name="write_file",
        func=write_to_file,
        description="Save content to file"
    )
]

# Create ReAct agent
agent = create_react_agent(
    llm=ChatOpenAI(model="gpt-4"),
    tools=tools,
    prompt=react_prompt
)

# Run autonomous task
result = agent.invoke({
    "input": "Research AI trends and summarize"
})
