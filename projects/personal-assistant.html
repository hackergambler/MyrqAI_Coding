Project: Personal AI Assistant

Build a personal assistant with persistent memory, note-taking, and tool use. Remembers your preferences across sessions.

What You Will Build

A fully functional personal assistant with persistent memory, tool use, and natural conversation. It remembers your preferences, takes notes, and uses the current date/time.

Python - personal assistant with memory and tools
from openai import OpenAI
from datetime import datetime
import json, os

client = OpenAI()
MEMORY_FILE = "assistant_memory.json"

def load_memory() -> dict:
    if os.path.exists(MEMORY_FILE):
        with open(MEMORY_FILE) as f: return json.load(f)
    return {"preferences": {}, "notes": []}

def save_memory(m): 
    with open(MEMORY_FILE, "w") as f: json.dump(m, f, indent=2)

def remember(key: str, value: str) -> str:
    m = load_memory(); m["preferences"][key] = value; save_memory(m)
    return f"Remembered: {key} = {value}"

def add_note(note: str) -> str:
    m = load_memory()
    m["notes"].append({"text": note, "time": datetime.now().isoformat()})
    save_memory(m); return f"Note saved: {note}"

def get_notes() -> str:
    m = load_memory()
    return "\n".join(f"- {n['text']} ({n['time'][:10]})" for n in m["notes"][-10:]) or "No notes."

TOOLS = [
    {"type":"function","function":{"name":"remember","description":"Store a user preference.",
     "parameters":{"type":"object","properties":{"key":{"type":"string"},"value":{"type":"string"}},"required":["key","value"]}}},
    {"type":"function","function":{"name":"add_note","description":"Save a note or reminder.",
     "parameters":{"type":"object","properties":{"note":{"type":"string"}},"required":["note"]}}},
    {"type":"function","function":{"name":"get_notes","description":"Retrieve saved notes.",
     "parameters":{"type":"object","properties":{}}}},
]
FNS = {"remember": remember, "add_note": add_note, "get_notes": get_notes}

memory = load_memory()
prefs = json.dumps(memory["preferences"]) if memory["preferences"] else "none yet"
SYSTEM = f"You are a personal assistant. User preferences: {prefs}. Today: {datetime.now():%A %B %d %Y}."

conversation = [{"role": "system", "content": SYSTEM}]

def chat(user_input: str) -> str:
    conversation.append({"role": "user", "content": user_input})
    while True:
        resp = client.chat.completions.create(
            model="gpt-4o", messages=conversation,
            tools=TOOLS, tool_choice="auto"
        )
        msg = resp.choices[0].message
        conversation.append(msg)
        if resp.choices[0].finish_reason == "stop":
            return msg.content
        for tc in msg.tool_calls or []:
            result = FNS[tc.function.name](**json.loads(tc.function.arguments))
            conversation.append({"role":"tool","tool_call_id":tc.id,"content":result})

# Interactive session
while True:
    user = input("You: ").strip()
    if user.lower() == "quit": break
    print(f"Assistant: {chat(user)}")
💡 Extend it

Add web_search(), draft_email(), set_reminder() tools. Each tool makes the assistant more capable without changing the core loop.
