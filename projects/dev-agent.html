Project: Coding Agent

Build an agent that reads files, writes code, runs tests, and iterates until they pass.

What You Will Build

An agent that reads your files, writes code, executes it in a sandboxed subprocess, interprets errors, and iterates until the code passes tests. The foundation of tools like Devin and GitHub Copilot Workspace.

Python - coding agent with file I/O and test execution
from openai import OpenAI
import json, subprocess, os, tempfile

client = OpenAI()

def read_file(path: str) -> str:
    try:
        with open(path) as f: return f.read()
    except FileNotFoundError: return f"File not found: {path}"

def write_file(path: str, content: str) -> str:
    with open(path, "w") as f: f.write(content)
    return f"Written {len(content)} chars to {path}"

def run_code(code: str) -> str:
    with tempfile.NamedTemporaryFile(mode="w", suffix=".py", delete=False) as f:
        f.write(code); tmp = f.name
    try:
        r = subprocess.run(["python3", tmp], capture_output=True, text=True, timeout=10)
        return f"EXIT {r.returncode}\nSTDOUT:\n{r.stdout[-1500:]}\nSTDERR:\n{r.stderr[-500:]}"
    except subprocess.TimeoutExpired: return "Error: timed out"
    finally: os.unlink(tmp)

def run_tests(test_file: str) -> str:
    r = subprocess.run(["python3", "-m", "pytest", test_file, "-v", "--tb=short"],
                       capture_output=True, text=True, timeout=30)
    return (r.stdout + r.stderr)[-3000:]

TOOLS = [
    {"type":"function","function":{"name":"read_file","description":"Read a file.",
     "parameters":{"type":"object","properties":{"path":{"type":"string"}},"required":["path"]}}},
    {"type":"function","function":{"name":"write_file","description":"Write content to a file.",
     "parameters":{"type":"object","properties":{"path":{"type":"string"},"content":{"type":"string"}},"required":["path","content"]}}},
    {"type":"function","function":{"name":"run_code","description":"Execute Python code.",
     "parameters":{"type":"object","properties":{"code":{"type":"string"}},"required":["code"]}}},
    {"type":"function","function":{"name":"run_tests","description":"Run pytest.",
     "parameters":{"type":"object","properties":{"test_file":{"type":"string"}},"required":["test_file"]}}},
]
FNS = {"read_file":read_file,"write_file":write_file,"run_code":run_code,"run_tests":run_tests}

def coding_agent(task: str, max_steps: int = 15) -> str:
    msgs = [
        {"role":"system","content":"You are an expert coding agent. Write code, test it, iterate."},
        {"role":"user","content":task}
    ]
    for step in range(max_steps):
        resp = client.chat.completions.create(
            model="gpt-4o", messages=msgs, tools=TOOLS, tool_choice="auto"
        )
        msg = resp.choices[0].message; msgs.append(msg)
        if resp.choices[0].finish_reason == "stop": return msg.content
        for tc in msg.tool_calls or []:
            args = json.loads(tc.function.arguments)
            result = FNS[tc.function.name](**args)
            print(f"  step {step+1}: {tc.function.name}({list(args.keys())})")
            msgs.append({"role":"tool","tool_call_id":tc.id,"content":str(result)})
    return "Max steps reached."

print(coding_agent("Write a Sieve of Eratosthenes function. Write pytest tests. Run them and ensure they pass."))
⚠️ Sandbox safety

Never run this agent on untrusted code. The subprocess runs with your user permissions. In production, use Docker containers or a proper sandbox like e2b.dev.
