Gemini Multi-turn Chat

Build stateful conversations using Gemini's ChatSession API, manage history, stream responses, and handle context limits.

ChatSession Basics

Python — stateful chat with history
import google.generativeai as genai

model = genai.GenerativeModel(
    "gemini-1.5-flash",
    system_instruction="You are a friendly Python tutor. Give short, clear answers."
)

# Start a chat session — it keeps history automatically
chat = model.start_chat(history=[])

def ask(question: str) -> str:
    response = chat.send_message(question)
    return response.text

print(ask("What is a Python decorator?"))
print(ask("Can you show me a real example?"))  # remembers context
print(ask("When should I use @functools.wraps?"))  # still in context

# Inspect history
for msg in chat.history:
    print(f"{msg.role}: {msg.parts[0].text[:80]}")
Python — pre-load history
# Start with existing conversation history
history = [
    {"role": "user", "parts": ["Hi, I'm learning Python."]},
    {"role": "model", "parts": ["Great! Python is a fantastic language to start with. What would you like to learn?"]},
]

chat = model.start_chat(history=history)
response = chat.send_message("Tell me about list comprehensions.")
print(response.text)

Streaming Chat

Python - stream chat responses
# Stream tokens as they arrive
chat = model.start_chat()
response = chat.send_message(
    "Write a 200-word story about a programmer who discovers their code is sentient.",
    stream=True
)

for chunk in response:
    print(chunk.text, end="", flush=True)

# Access final result after streaming
response.resolve()
print(f"\nTotal tokens: {response.usage_metadata.total_token_count}")

Async Chat for Web Apps

Python - async Gemini chat
import asyncio
import google.generativeai as genai

model = genai.GenerativeModel("gemini-1.5-flash")

async def chat_stream(messages: list[str]):
    chat = model.start_chat()
    for msg in messages:
        response = await chat.send_message_async(msg, stream=True)
        async for chunk in response:
            yield chunk.text

async def main():
    async for token in chat_stream([
        "What is asyncio?",
        "How is it different from threading?"
    ]):
        print(token, end="", flush=True)

asyncio.run(main())

Safety Settings

Python - configure safety filters
from google.generativeai.types import HarmCategory, HarmBlockThreshold

safety_settings = {
    HarmCategory.HARM_CATEGORY_HARASSMENT: HarmBlockThreshold.BLOCK_NONE,
    HarmCategory.HARM_CATEGORY_HATE_SPEECH: HarmBlockThreshold.BLOCK_MEDIUM_AND_ABOVE,
    HarmCategory.HARM_CATEGORY_SEXUALLY_EXPLICIT: HarmBlockThreshold.BLOCK_HIGH_AND_ABOVE,
    HarmCategory.HARM_CATEGORY_DANGEROUS_CONTENT: HarmBlockThreshold.BLOCK_MEDIUM_AND_ABOVE,
}

model = genai.GenerativeModel(
    "gemini-1.5-flash",
    safety_settings=safety_settings
)

# Check if content was blocked
response = model.generate_content("Your prompt here")
if response.candidates[0].finish_reason.name == "SAFETY":
    print("Blocked by safety filter")
    for rating in response.candidates[0].safety_ratings:
        print(f"  {rating.category.name}: {rating.probability.name}")
else:
    print(response.text)
