Gemini Function Calling

Give Gemini access to real-world tools: define functions, parse tool calls, execute code, and build complete agentic workflows.

Defining Tools

Python - tool definitions for Gemini
import google.generativeai as genai

# Define Python functions — Gemini reads the docstrings!
def get_weather(city: str, units: str = "celsius") -> dict:
    """Get current weather for a city.

    Args:
        city: City name, e.g. 'London' or 'New York'
        units: Temperature units - 'celsius' or 'fahrenheit'

    Returns:
        Dict with temperature, condition, humidity
    """
    # In production: call a real weather API
    return {"city": city, "temp": 18, "condition": "Cloudy", "humidity": 72}

def search_database(query: str, limit: int = 5) -> dict:
    """Search the product database for items matching a query.

    Args:
        query: Search term
        limit: Max results to return (1-20)
    """
    return {"results": [{"id": 1, "name": "Python Book", "price": 29.99}]}

# Pass functions directly — Gemini extracts schema from docstrings
model = genai.GenerativeModel(
    "gemini-1.5-flash",
    tools=[get_weather, search_database]
)
Python — complete function call loop
import json

TOOLS_MAP = {"get_weather": get_weather, "search_database": search_database}

def agent(prompt: str) -> str:
    chat = model.start_chat()
    response = chat.send_message(prompt)

    while True:
        # Check if Gemini wants to call a function
        fn_calls = [p for p in response.candidates[0].content.parts
                    if p.function_call.name]
        if not fn_calls:
            break

        # Execute all requested functions
        fn_results = []
        for part in fn_calls:
            fc = part.function_call
            fn = TOOLS_MAP.get(fc.name)
            result = fn(**dict(fc.args)) if fn else {"error": "unknown function"}
            fn_results.append(genai.protos.Part(
                function_response=genai.protos.FunctionResponse(
                    name=fc.name,
                    response={"result": result}
                )
            ))

        response = chat.send_message(fn_results)

    return response.text

print(agent("What's the weather in Tokyo and search for 'Python' products?"))
