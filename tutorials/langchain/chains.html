LangChain Chains

Build complex pipelines: sequential chains, transform chains, router chains, document chains, and custom LCEL runnables.

Sequential Chains

Chain multiple LLM calls where each output feeds into the next step.

Python — multi-step code review pipeline
from langchain_openai import ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import StrOutputParser
from langchain_core.runnables import RunnablePassthrough

llm = ChatOpenAI(model="gpt-4o-mini")
parser = StrOutputParser()

# Step 1: Find bugs
find_bugs = (
    ChatPromptTemplate.from_template("List all bugs in this code:\n{code}")
    | llm | parser
)

# Step 2: Fix the bugs (receives original code + bug list)
fix_bugs = (
    ChatPromptTemplate.from_template(
        "Original code:\n{code}\n\nBugs found:\n{bugs}\n\nWrite the fixed code:"
    )
    | llm | parser
)

# Step 3: Write tests for the fixed code
write_tests = (
    ChatPromptTemplate.from_template("Write pytest unit tests for:\n{fixed_code}")
    | llm | parser
)

# Connect them with LCEL
pipeline = (
    RunnablePassthrough.assign(bugs=find_bugs)
    | RunnablePassthrough.assign(fixed_code=lambda x: fix_bugs.invoke(x))
    | RunnablePassthrough.assign(tests=lambda x: write_tests.invoke({"fixed_code": x["fixed_code"]}))
)

result = pipeline.invoke({"code": """
def divide(a, b):
    return a / b  # Bug: no zero division check

def get_item(lst, idx):
    return lst[idx]  # Bug: no bounds check
"""})

print("=== BUGS ===")
print(result["bugs"])
print("\n=== FIXED ===")
print(result["fixed_code"])
print("\n=== TESTS ===")
print(result["tests"])

Document Chains

Process large documents by stuffing, refining, or map-reducing them.

Python — map-reduce summarization
from langchain.chains.combine_documents import create_stuff_documents_chain
from langchain.chains import create_retrieval_chain, MapReduceDocumentsChain
from langchain_core.documents import Document
from langchain.chains.mapreduce import MapReduceChain
from langchain.chains.summarize import load_summarize_chain

# Load your documents
docs = [
    Document(page_content="Chapter 1: Python was created by Guido van Rossum in 1991..."),
    Document(page_content="Chapter 2: Python's design philosophy emphasizes readability..."),
    Document(page_content="Chapter 3: Python 3.0 introduced many breaking changes in 2008..."),
]

# Map-reduce: summarize each doc, then combine summaries
chain = load_summarize_chain(llm, chain_type="map_reduce")
summary = chain.invoke(docs)
print(summary["output_text"])

# Refine: iteratively improve a running summary
refine_chain = load_summarize_chain(llm, chain_type="refine")
refined = refine_chain.invoke(docs)
print(refined["output_text"])

Custom LCEL Runnables

Python — custom runnable with RunnableLambda
from langchain_core.runnables import RunnableLambda, RunnableParallel
import time

# Wrap any Python function as a runnable
def preprocess(text: str) -> str:
    return text.strip().lower()[:1000]  # clean & truncate

def add_timestamp(text: str) -> dict:
    return {"text": text, "timestamp": time.time()}

clean = RunnableLambda(preprocess)
stamp = RunnableLambda(add_timestamp)

# Run two chains in PARALLEL and combine results
parallel = RunnableParallel(
    sentiment=(ChatPromptTemplate.from_template("Sentiment of: {text}") | llm | parser),
    summary=(ChatPromptTemplate.from_template("Summarize in 1 line: {text}") | llm | parser),
    keywords=(ChatPromptTemplate.from_template("Top 3 keywords from: {text}") | llm | parser),
)

full = clean | {"text": RunnablePassthrough()} | parallel
result = full.invoke("  LangChain makes it easy to build LLM applications!  ")
print(result)

Caching Chains

Python — in-memory and SQLite caching
from langchain_core.globals import set_llm_cache
from langchain_community.cache import InMemoryCache, SQLiteCache

# Option 1: in-memory (lost on restart)
set_llm_cache(InMemoryCache())

# Option 2: SQLite (persists across runs)
set_llm_cache(SQLiteCache(database_path="llm_cache.db"))

# Same chain — second call returns cached result instantly
chain = ChatPromptTemplate.from_template("Tell me a fact about {topic}") | llm | parser
print(chain.invoke({"topic": "Python"}))  # calls API
print(chain.invoke({"topic": "Python"}))  # from cache
