OpenAI Function Calling

Define tools, handle function calls, use parallel tool calls, and build real agents with OpenAI's function calling API.

Defining Tools

Python — tool definition
from openai import OpenAI
import json

client = OpenAI()

tools = [
    {
        "type": "function",
        "function": {
            "name": "search_products",
            "description": "Search the product catalog by keyword and optional category.",
            "parameters": {
                "type": "object",
                "properties": {
                    "query": {"type": "string", "description": "Search keyword"},
                    "category": {"type": "string", "enum": ["electronics", "clothing", "books"]},
                    "max_price": {"type": "number", "description": "Max price in USD"},
                    "limit": {"type": "integer", "default": 10}
                },
                "required": ["query"]
            }
        }
    },
    {
        "type": "function",
        "function": {
            "name": "get_order_status",

                "type": "object",
                "properties": {
                    "order_id": {"type": "string"}
                },
                "required": ["order_id"]

        }

Complete Tool Call Loop

Python — agentic tool call loop
def search_products(query, category=None, max_price=None, limit=10):
    # Simulated product DB
    products = [
        {"id": 1, "name": "Python Handbook", "category": "books", "price": 29.99},
        {"id": 2, "name": "Mechanical Keyboard", "category": "electronics", "price": 89.99},
    ]
    results = [p for p in products if query.lower() in p["name"].lower()]
    if category: results = [p for p in results if p["category"] == category]
    if max_price: results = [p for p in results if p["price"] <= max_price]
    return {"results": results[:limit]}

def get_order_status(order_id):
    return {"order_id": order_id, "status": "shipped", "eta": "2026-03-28"}

FN_MAP = {"search_products": search_products, "get_order_status": get_order_status}

def agent_loop(user_message: str) -> str:
    messages = [{"role": "user", "content": user_message}]

    while True:
        response = client.chat.completions.create(
            model="gpt-4o-mini",
            messages=messages,
            tools=tools,
            tool_choice="auto"
        )
        msg = response.choices[0].message
        messages.append(msg)

        if response.choices[0].finish_reason == "stop":
            return msg.content

        # Execute all requested tool calls
        for tc in msg.tool_calls or []:
            fn = FN_MAP[tc.function.name]
            args = json.loads(tc.function.arguments)
            result = fn(**args)
            messages.append({
                "role": "tool",
                "tool_call_id": tc.id,
                "content": json.dumps(result)
            })

print(agent_loop("Find me Python books under $50, and also check order ORD-12345"))

Structured Output (Strict Mode)

Python — guaranteed JSON schema output
from pydantic import BaseModel
from typing import List

class ExtractedTask(BaseModel):
    title: str
    priority: str  # high, medium, low
    due_date: str  # YYYY-MM-DD or "none"
    assignee: str

class TaskList(BaseModel):
    tasks: List[ExtractedTask]
    total_count: int

response = client.beta.chat.completions.parse(
    model="gpt-4o-mini",
    messages=[{
        "role": "user",
        "content": "Extract tasks from: Alice must finish the API by Friday (high priority). Bob review the PRs sometime next week."
    }],
    response_format=TaskList
)
task_list = response.choices[0].message.parsed
print(f"{task_list.total_count} tasks extracted")
for task in task_list.tasks:
    print(f"  [{task.priority}] {task.title} — {task.assignee} by {task.due_date}")
