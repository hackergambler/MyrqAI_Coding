OpenAI Embeddings

Generate embeddings with text-embedding-3 models, build semantic search, classify text, detect duplicates, and power RAG pipelines.

Generating Embeddings

Python — basic embedding generation
from openai import OpenAI
import numpy as np

client = OpenAI()

# Single text
response = client.embeddings.create(
    model="text-embedding-3-small",   # 1536 dims, cheaper
    input="Python is a great programming language"
)
vector = response.data[0].embedding
print(len(vector))  # 1536

# Batch (more efficient - up to 2048 strings)
texts = ["Hello world", "Machine learning", "Python functions"]
response = client.embeddings.create(
    model="text-embedding-3-large",    # 3072 dims, more accurate
    input=texts,
    dimensions=512                      # optionally reduce dims
)
vectors = [d.embedding for d in response.data]
print(f"Cost: ${response.usage.total_tokens / 1_000_000 * 0.13:.6f}")

Semantic Search from Scratch

Python — build a semantic search engine
import numpy as np
from openai import OpenAI

client = OpenAI()

class SemanticSearchEngine:
    def __init__(self, model="text-embedding-3-small"):
        self.model = model
        self.documents = []
        self.embeddings = []

    def _embed(self, texts: list[str]) -> np.ndarray:
        res = client.embeddings.create(model=self.model, input=texts)
        return np.array([d.embedding for d in res.data])

    def add_documents(self, docs: list[str]):
        vecs = self._embed(docs)
        self.documents.extend(docs)
        if len(self.embeddings) == 0:
            self.embeddings = vecs
        else:
            self.embeddings = np.vstack([self.embeddings, vecs])

    def search(self, query: str, k: int = 5) -> list[tuple[float, str]]:
        q_vec = self._embed([query])
        # Cosine similarity via dot product on L2-normalized vectors
        norms = np.linalg.norm(self.embeddings, axis=1, keepdims=True)
        normed = self.embeddings / norms
        q_normed = q_vec / np.linalg.norm(q_vec)
        scores = (normed @ q_normed.T).flatten()
        top_k = np.argsort(scores)[::-1][:k]
        return [(scores[i], self.documents[i]) for i in top_k]

# Demo
engine = SemanticSearchEngine()
engine.add_documents([
    "Python list comprehensions create concise loops",
    "FastAPI builds REST APIs with type hints",
    "NumPy arrays are faster than Python lists",
    "Django includes an ORM and admin panel",
    "asyncio enables concurrent I/O operations",
])

results = engine.search("How do I make async HTTP requests?", k=2)
for score, doc in results:
    print(f"[{score:.3f}] {doc}")

Text Classification with Embeddings

Python — zero-shot classification
import numpy as np

def classify_zeroshot(text: str, labels: list[str]) -> str:
    """Classify text into one of the given labels using embeddings."""
    all_texts = [text] + labels
    res = client.embeddings.create(model="text-embedding-3-small", input=all_texts)
    vecs = np.array([d.embedding for d in res.data])

    text_vec = vecs[0]
    label_vecs = vecs[1:]

    # Cosine similarity
    scores = []
    for lv in label_vecs:
        score = np.dot(text_vec, lv) / (np.linalg.norm(text_vec) * np.linalg.norm(lv))
        scores.append(score)

    best = np.argmax(scores)
    return labels[best], scores

# Test it
text = "My laptop keeps crashing after the latest update"
labels = ["billing", "technical_support", "account", "sales"]
category, scores = classify_zeroshot(text, labels)
print(f"Category: {category}")
for label, score in zip(labels, scores):
    print(f"  {label}: {score:.3f}")

Duplicate Detection

Python — find near-duplicate content
from itertools import combinations
import numpy as np

def find_duplicates(texts: list[str], threshold: float = 0.95) -> list[tuple]:
    res = client.embeddings.create(model="text-embedding-3-small", input=texts)
    vecs = np.array([d.embedding for d in res.data])

    # Normalize
    norms = np.linalg.norm(vecs, axis=1, keepdims=True)
    vecs = vecs / norms

    # Compare all pairs
    duplicates = []
    for i, j in combinations(range(len(texts)), 2):
        sim = np.dot(vecs[i], vecs[j])
        if sim >= threshold:
            duplicates.append((i, j, sim, texts[i][:50], texts[j][:50]))
    return duplicates

texts = [
    "How do I install Python?",
    "What's the best way to install Python on Windows?",
    "Python installation guide",
    "What is machine learning?",
]
dupes = find_duplicates(texts, threshold=0.85)
print(f"Found {len(dupes)} near-duplicates")
