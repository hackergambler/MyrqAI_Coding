Tokens, Models & Cost Optimization

How tokenization works, how to count tokens before API calls, choose the right model, and minimize costs in production.

What is a Token?

Tokens are chunks of text — roughly 4 characters or 0.75 words in English. "Hello world" is 2 tokens. The entire works of Shakespeare is ~900,000 tokens.

Python — count tokens with tiktoken
import tiktoken

def count_tokens(text: str, model: str = "gpt-4o") -> int:
    enc = tiktoken.encoding_for_model(model)
    return len(enc.encode(text))

def count_message_tokens(messages: list[dict], model: str = "gpt-4o") -> int:
    """Count tokens in a messages array, including overhead."""
    enc = tiktoken.encoding_for_model(model)
    tokens_per_message = 3    # overhead per message
    tokens_per_name = 1       # overhead for name field
    total = 3                  # reply priming
    for msg in messages:
        total += tokens_per_message
        for key, val in msg.items():
            total += len(enc.encode(str(val)))
            if key == "name": total += tokens_per_name
    return total

# Test it
msgs = [
    {"role": "system", "content": "You are a coding assistant."},
    {"role": "user", "content": "What is a Python generator?"}
]
print(f"Estimated tokens: {count_message_tokens(msgs)}")

Model Comparison

  • gpt-4o — Flagship multimodal model. Best quality, vision capable. $2.50/1M in · $10/1M out
  • gpt-4o-mini — Affordable, fast. Great for most tasks. $0.15/1M in · $0.60/1M out
  • o1 — Extended thinking for complex reasoning. $15/1M in · $60/1M out
  • o3-mini — Fast reasoning model. $1.10/1M in · $4.40/1M out
  • gpt-3.5-turbo — Legacy. Use gpt-4o-mini instead.
Python — auto model selection by task
def select_model(task_type: str, quality: str = "balanced") -> str:
    """Select optimal model based on task and quality requirement."""
    MODELS = {
        "simple": {"fast": "gpt-4o-mini", "balanced": "gpt-4o-mini", "best": "gpt-4o"},
        "reasoning": {"fast": "o3-mini", "balanced": "o3-mini", "best": "o1"},
        "vision": {"fast": "gpt-4o-mini", "balanced": "gpt-4o", "best": "gpt-4o"},
        "code": {"fast": "gpt-4o-mini", "balanced": "gpt-4o", "best": "o1"},
    }
    return MODELS.get(task_type, MODELS["simple"])[quality]

print(select_model("code", "balanced"))  # gpt-4o
print(select_model("simple", "fast"))     # gpt-4o-mini

Cost Tracking in Production

Python — per-call and cumulative cost tracking
from dataclasses import dataclass, field
from threading import Lock

PRICING = {
    "gpt-4o":         {"in": 2.50, "out": 10.0},
    "gpt-4o-mini":    {"in": 0.15, "out": 0.60},
    "o1":             {"in": 15.0, "out": 60.0},
    "text-embedding-3-small": {"in": 0.02, "out": 0},
}

class CostTracker:
    def __init__(self):
        self.total_cost = 0.0
        self.calls = 0
        self._lock = Lock()

    def track(self, model: str, in_tokens: int, out_tokens: int) -> float:
        rates = PRICING.get(model, {"in": 0, "out": 0})
        cost = (in_tokens * rates["in"] + out_tokens * rates["out"]) / 1_000_000
        with self._lock:
            self.total_cost += cost
            self.calls += 1
        return cost

    def report(self):
        print(f"Total: ${self.total_cost:.6f} over {self.calls} calls")

tracker = CostTracker()

def tracked_call(**kwargs):
    resp = client.chat.completions.create(**kwargs)
    cost = tracker.track(
        kwargs["model"],
        resp.usage.prompt_tokens,
        resp.usage.completion_tokens
    )
    print(f"  Call cost: ${cost:.6f}")
    return resp

tracked_call(model="gpt-4o-mini", messages=[{"role": "user", "content": "Hello"}])
tracker.report()
