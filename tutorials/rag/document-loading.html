Document Loading for RAG

Load documents from any source into your RAG pipeline: PDFs, websites, Word docs, Notion, GitHub, databases, and custom loaders.

PDF Loading

Python — PDF loading options
from langchain_community.document_loaders import (
    PyPDFLoader,
    PDFMinerLoader,
    UnstructuredPDFLoader,
)

# Option 1: PyPDF (fast, basic)
loader = PyPDFLoader("document.pdf")
pages = loader.load()
print(f"{len(pages)} pages loaded")
print(pages[0].page_content[:500])

# Option 2: PDFMiner (better text extraction)
loader2 = PDFMinerLoader("document.pdf")
docs = loader2.load()

# Option 3: Unstructured (best, handles tables/images)
loader3 = UnstructuredPDFLoader("document.pdf", mode="elements")
elements = loader3.load()
for el in elements[:5]:
    print(el.metadata.get("category"), el.page_content[:100])

# Load all PDFs in a directory
from langchain_community.document_loaders import DirectoryLoader
dir_loader = DirectoryLoader("./docs", glob="**/*.pdf", loader_cls=PyPDFLoader)
all_docs = dir_loader.load()
print(f"Loaded {len(all_docs)} documents")

Web & URL Loading

Python — web scraping loaders
from langchain_community.document_loaders import (
    WebBaseLoader,
    RecursiveUrlLoader,
    SitemapLoader,
)
import bs4

# Single URL
loader = WebBaseLoader(
    web_paths=["https://docs.python.org/3/library/asyncio.html"],
    bs_kwargs={"parse_only": bs4.SoupStrainer("article")}
)
docs = loader.load()

# Crawl an entire docs site recursively
recursive = RecursiveUrlLoader(
    url="https://docs.python.org/3/",
    max_depth=2,
    extractor=lambda x: bs4.BeautifulSoup(x, "html.parser").get_text()
)
crawled = recursive.load()
print(f"Crawled {len(crawled)} pages")

# Load from sitemap
sitemap = SitemapLoader("https://example.com/sitemap.xml", filter_urls=["docs"])
sitemap_docs = sitemap.load()

Office Documents

Python — Word, Excel, and PowerPoint
from langchain_community.document_loaders import (
    Docx2txtLoader,

# Word documents
word_loader = Docx2txtLoader("report.docx")

Code & GitHub

Python — load source code for RAG
from langchain_community.document_loaders import (
    GenericLoader,
    GitLoader,
    PythonLoader,
)
from langchain_community.document_loaders.parsers import LanguageParser
from langchain_text_splitters import Language

# Load Python files with code-aware splitting
loader = GenericLoader.from_filesystem(
    "./my_project",
    glob="**/*.py",
    suffixes=[".py"],
    parser=LanguageParser(language=Language.PYTHON, parser_threshold=500),
)
code_docs = loader.load()

# Clone and load a GitHub repo
git_loader = GitLoader(
    clone_url="https://github.com/langchain-ai/langchain",
    repo_path="./langchain_repo",
    branch="master",
    file_filter=lambda f: f.endswith(".py")
)
repo_docs = git_loader.load()
print(f"Loaded {len(repo_docs)} Python files")

Custom Loader

Python — build your own document loader
from langchain_core.document_loaders import BaseLoader
from langchain_core.documents import Document
from typing import Iterator
import json, pathlib

class JSONLLoader(BaseLoader):
    """Load documents from a JSON Lines file."""
    def __init__(self, file_path: str, content_key: str, metadata_keys: list = None):
        self.file_path = pathlib.Path(file_path)
        self.content_key = content_key
        self.metadata_keys = metadata_keys or []

    def lazy_load(self) -> Iterator[Document]:
        with self.file_path.open() as f:
            for i, line in enumerate(f):
                obj = json.loads(line.strip())
                metadata = {k: obj.get(k) for k in self.metadata_keys}
                metadata["source"] = str(self.file_path)
                metadata["line"] = i
                yield Document(page_content=obj.get(self.content_key, ""), metadata=metadata)

# Use it
loader = JSONLLoader("reviews.jsonl", content_key="text", metadata_keys=["rating", "product_id"])
docs = list(loader.lazy_load())
print(f"Loaded {len(docs)} reviews")
