Tool Use & Function Calling

Give Claude the ability to call your code. Define tools, parse tool requests, execute functions, and build multi-step agentic workflows.

1. How Tool Use Works

Tool use is a 3-step loop: (1) Claude decides to call a tool and returns a tool_use block. (2) Your code executes the tool. (3) You send the result back as a tool_result message. Repeat until Claude returns end_turn.

ℹ️ Stop Reason

When Claude wants to call a tool, stop_reason is "tool_use". When it's done, stop_reason is "end_turn". Always check this.

2. Defining Tools (JSON Schema)

Python — tool definition structure
tools = [
    {
        "name": "get_weather",
        "description": """Get current weather for a city.
Returns temperature in Celsius and conditions.""",
        "input_schema": {
            "type": "object",
            "properties": {
                "city": {
                    "type": "string",
                    "description": "City name, e.g. 'London' or 'New York'"
                },
                "units": {
                    "type": "string",
                    "enum": ["celsius", "fahrenheit"],
                    "description": "Temperature unit"
                }
            },
            "required": ["city"]
        }
    },
    {
        "name": "calculate",
        "description": "Evaluate a mathematical expression safely.",
        "input_schema": {
            "type": "object",
            "properties": {
                "expression": {
                    "type": "string",
                    "description": "Math expression, e.g. '2 * (3 + 4)'"
                }
            },
            "required": ["expression"]
        }
    }
]
💡 Tool Descriptions Matter

Claude reads your tool descriptions to decide when and how to call them. Be specific. Include units, formats, and limitations. A good description is worth more than a complex schema.

3. Basic Tool Call Loop

Python — complete tool call loop
import json, anthropic

client = anthropic.Anthropic()

# Your real tool implementations
def get_weather(city: str, units: str = "celsius") -> dict:
    # In production: call a real weather API
    return {"city": city, "temp": 22, "condition": "Partly cloudy", "units": units}

def calculate(expression: str) -> dict:
    try:
        result = eval(expression, {"__builtins__": {}})  # safe eval
        return {"result": result}
    except Exception as e:
        return {"error": str(e)}

TOOLS_MAP = {"get_weather": get_weather, "calculate": calculate}

def run_agent(user_message: str) -> str:
    messages = [{"role": "user", "content": user_message}]

    while True:
        response = client.messages.create(
            model="claude-opus-4-5",
            max_tokens=1024,
            tools=tools,
            messages=messages
        )

        # Add assistant response to history
        messages.append({"role": "assistant", "content": response.content})

        if response.stop_reason == "end_turn":
            # Done — extract text response
            for block in response.content:
                if hasattr(block, "text"):
                    return block.text
            return ""

        if response.stop_reason == "tool_use":
            tool_results = []
            for block in response.content:
                if block.type == "tool_use":
                    fn = TOOLS_MAP.get(block.name)
                    if fn:
                        result = fn(**block.input)
                    else:
                        result = {"error": f"Unknown tool: {block.name}"}
                    tool_results.append({
                        "type": "tool_result",
                        "tool_use_id": block.id,
                        "content": json.dumps(result)
                    })
            messages.append({"role": "user", "content": tool_results})

print(run_agent("What's the weather in Tokyo? Also, what is 144 * 12?"))

4. Parallel Tool Calls

Claude can request multiple tools at once. Your code should run them in parallel for efficiency.

Python — parallel execution with asyncio
import asyncio, json, anthropic

async def execute_tools_parallel(tool_blocks: list) -> list:
    async def run_one(block):
        fn = TOOLS_MAP.get(block.name)
        result = fn(**block.input) if fn else {"error": "not found"}
        return {
            "type": "tool_result",
            "tool_use_id": block.id,
            "content": json.dumps(result)
        }
    return await asyncio.gather(*[run_one(b) for b in tool_blocks if b.type == "tool_use"])

5. Real-World Tool Examples

Python — database query tool
import sqlite3

def query_database(sql: str, params: list = None) -> dict:
    """Execute a read-only SQL query and return results."""
    if not sql.strip().upper().startswith("SELECT"):
        return {"error": "Only SELECT queries are allowed"}
    try:
        conn = sqlite3.connect("data.db")
        conn.row_factory = sqlite3.Row
        cursor = conn.execute(sql, params or [])
        rows = [dict(row) for row in cursor.fetchmany(100)]
        return {"rows": rows, "count": len(rows)}
    except Exception as e:
        return {"error": str(e)}
    finally:
        conn.close()

db_tool = {
    "name": "query_database",
    "description": "Run a SELECT query on the products database.",
    "input_schema": {
        "type": "object",
        "properties": {
            "sql": {"type": "string", "description": "SELECT SQL query"},
            "params": {"type": "array", "items": {"type": "string"}}
        },
        "required": ["sql"]
    }
}
Python — web scraper tool
import httpx
from bs4 import BeautifulSoup

def fetch_webpage(url: str, selector: str = "body") -> dict:
    """Fetch and extract text from a webpage."""
    try:
        r = httpx.get(url, timeout=10, headers={"User-Agent": "Mozilla/5.0"})
        soup = BeautifulSoup(r.text, "html.parser")
        el = soup.select_one(selector) or soup.body
        text = el.get_text(separator="\n", strip=True)[:5000]  # limit
        return {"url": url, "content": text, "chars": len(text)}
    except Exception as e:
        return {"error": str(e)}

6. Error Handling in Tool Results

Python — returning errors as tool results
# Never raise exceptions inside tool logic.
# Always return a dict with an "error" key.
# Claude will read the error and decide what to do.

def safe_tool(param: str) -> dict:
    try:
        # ... do work ...
        return {"result": "success"}
    except ValueError as e:
        return {"error": f"Invalid input: {e}", "retry_possible": True}
    except Exception as e:
        return {"error": str(e), "retry_possible": False}

# When passing back to Claude, mark it as an error
tool_results.append({
    "type": "tool_result",
    "tool_use_id": block.id,
    "content": json.dumps(result),
    "is_error": "error" in result
})

7. Building a Simple Research Agent

Python — research agent with web + calculator tools
import json, httpx, anthropic
from bs4 import BeautifulSoup

client = anthropic.Anthropic()

def web_search(query: str) -> dict:
    r = httpx.get(f"https://api.duckduckgo.com/?q={query}&format=json", timeout=8)
    data = r.json()
    abstract = data.get("AbstractText", "")
    return {"summary": abstract[:500] if abstract else "No results found."}

def calculate(expr: str) -> dict:
    try: return {"result": eval(expr, {"__builtins__": {}})}
    except Exception as e: return {"error": str(e)}

TOOLS = [
    {"name": "web_search", "description": "Search the web for factual info.",
     "input_schema": {"type": "object", "properties": {"query": {"type": "string"}}, "required": ["query"]}},
    {"name": "calculate", "description": "Evaluate math expressions.",
     "input_schema": {"type": "object", "properties": {"expr": {"type": "string"}}, "required": ["expr"]}},
]
FN = {"web_search": web_search, "calculate": calculate}

def research_agent(question: str, max_steps: int = 10) -> str:
    msgs = [{"role": "user", "content": question}]
    for _ in range(max_steps):
        resp = client.messages.create(model="claude-opus-4-5", max_tokens=1024, tools=TOOLS, messages=msgs)
        msgs.append({"role": "assistant", "content": resp.content})
        if resp.stop_reason == "end_turn":
            return next((b.text for b in resp.content if hasattr(b, "text")), "")
        results = [{"type": "tool_result", "tool_use_id": b.id, "content": json.dumps(FN[b.name](**b.input))}
                   for b in resp.content if b.type == "tool_use"]
        msgs.append({"role": "user", "content": results})
    return "Max steps reached."

print(research_agent("What is the population of Japan? Calculate how many people that is per square km (area 377,975 km²)."))
