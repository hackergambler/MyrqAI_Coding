Multi-Agent Systems

Coordinate multiple specialized AI agents: orchestrator-worker patterns, agent handoffs, parallel execution, and building a complete research team.

Orchestrator-Worker Pattern

Python - orchestrator delegates to specialist agents
from openai import OpenAI
from typing import Callable
import json

client = OpenAI()

class Agent:
    def __init__(self, name: str, system: str, tools: list = None):
        self.name = name
        self.system = system
        self.tools = tools or []

    def run(self, task: str) -> str:
        resp = client.chat.completions.create(
            model="gpt-4o-mini",
            messages=[
                {"role": "system", "content": self.system},
                {"role": "user", "content": task}
            ]
        )
        return resp.choices[0].message.content

# Specialist agents
researcher = Agent("Researcher", "You find and summarize factual information concisely.")
writer = Agent("Writer", "You turn research notes into polished, engaging prose.")
critic = Agent("Critic", "You review content for accuracy, clarity, and quality. Be specific.")

# Orchestrator coordinates them
orchestrator = Agent("Orchestrator", """You coordinate a team of specialists.
When given a topic, output a JSON plan: 
{"steps": [{"agent": "Researcher|Writer|Critic", "task": "..."}]}""")

def run_pipeline(topic: str) -> str:
    # Get plan from orchestrator
    plan_raw = orchestrator.run(f"Create a plan to write a blog post about: {topic}")
    plan = json.loads(plan_raw)

    context = f"Topic: {topic}"
    agents_map = {"Researcher": researcher, "Writer": writer, "Critic": critic}

    for step in plan["steps"]:
        agent = agents_map[step["agent"]]
        task = f"{step['task']}\n\nContext so far:\n{context}"
        result = agent.run(task)
        context += f"\n\n[{agent.name}]:\n{result}"
        print(f"✓ {agent.name} complete")

    return context

result = run_pipeline("The future of AI agents in software development")
print(result)

Parallel Agent Execution

Python - run agents in parallel with asyncio
import asyncio
from openai import AsyncOpenAI

async_client = AsyncOpenAI()

async def agent_task(name: str, system: str, task: str) -> tuple[str, str]:
    resp = await async_client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "system", "content": system}, {"role": "user", "content": task}]
    )
    return name, resp.choices[0].message.content

async def parallel_research(topic: str) -> dict:
    tasks = [
        agent_task("Market Analyst", "Analyze market trends and business impact.", topic),
        agent_task("Tech Expert", "Explain technical implementation details.", topic),
        agent_task("Risk Assessor", "Identify risks and mitigation strategies.", topic),
    ]
    results = await asyncio.gather(*tasks)
    return dict(results)

# Run all agents in parallel — 3x faster than sequential
results = asyncio.run(parallel_research("Implementing LLMs in healthcare"))
for agent_name, output in results.items():
    print(f"\n=== {agent_name} ===\n{output}")
