Agent Tool Design

Great tools make great agents. Learn to design robust, safe, and descriptive tools that agents can reliably call, with proper error handling and security guards.

Tool Design Principles

  • Single responsibility — one tool does one thing clearly
  • Descriptive names — search_products_by_category not search
  • Rich descriptions — tell the agent when NOT to use the tool
  • Always return a value — never raise exceptions, return {error: ...}
  • Idempotent reads — GET operations should be safe to retry
  • Confirm before writes — destructive actions need human confirmation
Python - well-designed tool registry
from typing import Any
from dataclasses import dataclass
from functools import wraps
import logging, time

logger = logging.getLogger("agent.tools")

def safe_tool(func):
    """Decorator: catch all exceptions, log calls, enforce timeout."""
    @wraps(func)
    def wrapper(*args, **kwargs) -> dict:
        start = time.time()
        try:
            result = func(*args, **kwargs)
            elapsed = time.time() - start
            logger.info(f"{func.__name__} OK in {elapsed:.2f}s")
            return result if isinstance(result, dict) else {"result": result}
        except Exception as e:
            logger.error(f"{func.__name__} FAILED: {e}")
            return {"error": str(e), "tool": func.__name__}
    return wrapper

@safe_tool
def get_stock_price(ticker: str) -> dict:
    """Get the current stock price for a ticker symbol.
    
    Use this when the user asks about stock prices or market values.
    Do NOT use for crypto, forex, or commodities - those have separate tools.
    
    Args:
        ticker: Stock ticker symbol, e.g. 'AAPL', 'GOOGL', 'MSFT'
    
    Returns:
        price (float), currency (str), change_pct (float), market_status (str)
    """
    # Simulated — replace with real API call
    prices = {"AAPL": 195.50, "GOOGL": 175.20, "MSFT": 415.80}
    if ticker.upper() not in prices:
        return {"error": f"Unknown ticker: {ticker}. Try AAPL, GOOGL, or MSFT."}
    return {"ticker": ticker.upper(), "price": prices[ticker.upper()], "currency": "USD"}

@safe_tool
def send_email(to: str, subject: str, body: str, confirm: bool = False) -> dict:

Input Validation

Python — strict input validation with Pydantic
from pydantic import BaseModel, Field, validator
from typing import Optional
import json

class SearchInput(BaseModel):
    query: str = Field(..., min_length=1, max_length=200)
    max_results: int = Field(default=10, ge=1, le=50)
    category: Optional[str] = Field(default=None)

    @validator("query")
    def sanitize_query(cls, v):
        # Remove SQL injection attempts
        forbidden = ["DROP", "DELETE", "INSERT", "--"]
        if any(f.lower() in v.lower() for f in forbidden):
            raise ValueError("Invalid query")
        return v.strip()

def validated_tool(tool_name: str, raw_args: str) -> dict:
    """Validate agent-provided JSON before executing tool."""
    try:
        args = json.loads(raw_args)
        validated = SearchInput(**args)
        return {"validated": validated.dict()}
    except Exception as e:
        return {"error": f"Invalid args: {e}"}
