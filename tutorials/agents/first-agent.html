Your First Complete Agent

We'll build a research agent from scratch using OpenAI function calling. It can search the web, do math, and give a final answer. No frameworks — raw Python so you understand every piece.

Python - complete first agent
from openai import OpenAI
import json, httpx

client = OpenAI()

# ---- TOOLS ----
def web_search(query: str) -> str:
    r = httpx.get(
        f"https://api.duckduckgo.com/?q={{query}}&format=json",
        timeout=8
    )
    data = r.json()
    abstract = data.get("AbstractText", "")
    return abstract[:500] if abstract else "No results found."

def calculate(expression: str) -> str:
    try:
        result = eval(expression, {{"__builtins__": {{}}}})
        return str(result)
    except Exception as e:
        return f"Error: {{e}}"

TOOLS_DEF = [
    {{"type": "function", "function": {{
        "name": "web_search",
        "description": "Search the web for current factual information.",
        "parameters": {{"type": "object",
            "properties": {{"query": {{"type": "string"}}}},
            "required": ["query"]}}
    }}}}  ,
    {{"type": "function", "function": {{
        "name": "calculate",
        "description": "Evaluate a math expression. E.g. '2 ** 10' or '144 / 12'.",
        "parameters": {{"type": "object",
            "properties": {{"expression": {{"type": "string"}}}},
            "required": ["expression"]}}
    }}}}
]

TOOL_FNS = {{"web_search": web_search, "calculate": calculate}}

# ---- AGENT LOOP ----
def run_agent(goal: str, max_steps: int = 10) -> str:
    messages = [
        {{"role": "system", "content": "You are a research agent. Use tools to find information and answer questions accurately."}},
        {{"role": "user", "content": goal}}
    ]
    
    for step in range(max_steps):
        resp = client.chat.completions.create(
            model="gpt-4o-mini", messages=messages,
            tools=TOOLS_DEF, tool_choice="auto"
        )
        msg = resp.choices[0].message
        messages.append(msg)
        
        if resp.choices[0].finish_reason == "stop":
            return msg.content  # done
        
        for tc in msg.tool_calls or []:
            fn = TOOL_FNS[tc.function.name]
            args = json.loads(tc.function.arguments)
            result = fn(**args)
            print(f"  [{tc.function.name}] {{args}} → {{result[:80]}}")
            messages.append({{
                "role": "tool",
                "tool_call_id": tc.id,
                "content": result
            }})
    
    return "Max steps reached."

# ---- RUN IT ----
print(run_agent("What is the population of Japan? Calculate how many people per square km (area: 377975 km2)"))
💡 Max steps guard

Always set a maximum step limit. Without it, a buggy agent can loop forever, burning tokens and money.
